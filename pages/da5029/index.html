# JavaFX Button

提示

JavaFX按钮控件由javafx.scene.control.Button类表示。按钮是可以控制应用程序行为的组件。每当单击按钮时都会生成一个事件。

# 示例一

package code.javafx.JavaFXUI;

import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.layout.StackPane;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2024/7/5 19:36
 */
public class JavaFXButton extends Application {
    @Override
    public void start(Stage primaryStage) throws Exception {
        StackPane root = new StackPane();
        Button btn = new Button("This is a button");
        Scene scene = new Scene(root, 300, 300);

        root.getChildren().add(btn);
        primaryStage.setScene(scene);
        primaryStage.setTitle("Button Class Example");
        primaryStage.show();

    }

    public static void main(String[] args) {
        launch(args);
    }
}

# 按钮事件

package code.javafx.JavaFXUI;

import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.image.Image;
import javafx.scene.image.ImageView;
import javafx.scene.layout.StackPane;
import javafx.stage.Stage;

import java.io.FileInputStream;

/**
 * @author :ghgeng
 * @date 2024/7/5 19:36
 */
public class JavaFXButton extends Application {
    @Override
    public void start(Stage primaryStage) throws Exception {
        StackPane root = new StackPane();

        FileInputStream input = new FileInputStream("C:\\Users\\11476\\Desktop\\QQ截图20240705195028.png");
        Image image = new Image(input);
        ImageView img = new ImageView(image);

        Button btn = new Button("buton");
        btn.setGraphic(img);
        btn.setWrapText(true);
        // 添加事件处理
        btn.setOnAction(e -> {
            System.out.println("Button clicked");
        });
        Scene scene = new Scene(root, 300, 300);

        root.getChildren().add(btn);
        primaryStage.setScene(scene);
        primaryStage.setTitle("Button Class Example");
        primaryStage.show();

    }

    public static void main(String[] args) {
        launch(args);
    }
}

# 按钮特效

package code.javafx.JavaFXUI;

import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.effect.DropShadow;
import javafx.scene.image.Image;
import javafx.scene.image.ImageView;
import javafx.scene.layout.StackPane;
import javafx.stage.Stage;

import java.io.FileInputStream;

/**
 * @author :ghgeng
 * @date 2024/7/5 19:36
 */
public class JavaFXButton extends Application {
    @Override
    public void start(Stage primaryStage) throws Exception {
        StackPane root = new StackPane();

        FileInputStream input = new FileInputStream("C:\\Users\\11476\\Desktop\\QQ截图20240705195028.png");
        Image image = new Image(input);
        ImageView img = new ImageView(image);

        Button btn = new Button();
        btn.setGraphic(img);
        btn.setWrapText(true);

        // 阴影特效
        DropShadow dropShadow = new DropShadow();
        btn.setEffect(dropShadow);
        
        Scene scene = new Scene(root, 300, 300);

        root.getChildren().add(btn);
        primaryStage.setScene(scene);
        primaryStage.setTitle("Button Class Example");
        primaryStage.show();

    }

    public static void main(String[] args) {
        launch(args);
    }
}
