Java JScrollBar

# JScrollBar

JScrollbar类的对象用于添加水平和垂直滚动条。它是一个滚动条的实现。它继承JComponent类。JScrollbar类的对象用于添加水平和垂直滚动条。它是一个滚动条的实现。它继承JComponent类。

# 常用构造器:

Constructor Description
JScrollBar() 创建带有初始值的垂直滚动条。
JScrollBar(int orientation) 创建具有指定方向和初始值的滚动条。
JScrollBar(int orientation, int value, int extent, int min, int max) 创建一个具有指定方向、值、范围、最小值和最大值的滚动条。

# 示例代码1：

package code.swing.learn;

import javax.swing.*;

/**
 * @author :ghgeng
 * @date 2023/7/12 21:53
 */
public class JScrollBarTest {

    public static void main(String[] args) {
        JFrame j = new JFrame("JScrollBarTest");

        JScrollBar jScrollBar = new JScrollBar();
        jScrollBar.setBounds(50, 50, 100, 300);

        j.add(jScrollBar);

        // 显示窗口
        j.setSize(400, 400);
        j.setLayout(null);
        j.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        j.setLocationRelativeTo(null);
        j.setVisible(true);
    }
}

# 示例代码2： AdjustmentListener()

package code.swing.learn;

import javax.swing.*;

/**
 * @author :ghgeng
 * @date 2023/7/12 21:53
 */
public class JScrollBarTest {

    public static void main(String[] args) {
        JFrame j = new JFrame("JScrollBarTest");

        JLabel jLabel = new JLabel();
        jLabel.setBounds(20, 20, 100, 30);

        JScrollBar jScrollBar = new JScrollBar();
        jScrollBar.setBounds(50, 50, 100, 300);
        jScrollBar.addAdjustmentListener( e ->{
            jLabel.setText("滑动位置：" +  e.getValue());
        });

        j.add(jScrollBar);
        j.add(jLabel);
        // 显示窗口
        j.setSize(400, 400);
        j.setLayout(null);
        j.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        j.setLocationRelativeTo(null);
        j.setVisible(true);
    }
}
