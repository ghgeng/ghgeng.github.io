Java JLayeredPane

# JLayeredPane

JLayeredPane类用于为摆动容器添加深度。它用于为定位组件提供第三维度，并将深度范围划分为几个不同的层。

# 常用方法：

Method Description
int getIndexOf(Component c) 它用于返回指定组件的索引。
int getLayer(Component c) 它用于返回指定组件的层属性。
int getPosition(Component c) 它用于返回组件在其层中的相对位置。

# 示例代码1：

package code.swing.learn;

import javax.swing.*;
import java.awt.*;

/**
 * @author :ghgeng
 * @date 2023/7/20 16:50
 */
public class JLayeredPaneTest {

    public static void main(String[] args) {
        JFrame jFrame = new JFrame();
        // 获取分层窗口
        JLayeredPane layeredPane = jFrame.getLayeredPane();

        JButton jButton1 = new JButton("A");
        jButton1.setBackground(Color.red);
        jButton1.setBounds(10, 10, 100, 30);
        jFrame.add(jButton1);

        JButton jButton2 = new JButton("B");
        jButton2.setBackground(Color.pink);
        jButton2.setBounds(30, 30, 100, 30);
        jFrame.add(jButton1);

        JButton jButton3 = new JButton("C");
        jButton3.setBounds(50, 50, 100, 30);
        jButton3.setBackground(Color.green);
        jFrame.add(jButton1);

        layeredPane.add(jButton1 ,new Integer(1));
        layeredPane.add(jButton2 , new Integer(2));
        layeredPane.add(jButton3 , new Integer(3));

        jFrame.setTitle("JLayeredPaneTest");
        jFrame.setLocationRelativeTo(null);
        jFrame.setVisible(true);
        jFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        jFrame.setSize(400, 400);
    }
}
