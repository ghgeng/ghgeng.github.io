JavaFX ToolBar

# ToolBar

提示

在JavaFX中，ToolBar是一个用于放置多个控件（如按钮、文本字段等）的水平或垂直容器，这些控件通常用于提供应用程序的工具栏功能。以下是一个简单的ToolBar代码示例，展示了如何创建一个包含几个按钮的ToolBar，并将其添加到场景和舞台中。

# 代码示例

package code.javafx.learn.JavaFX布局;

import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.control.ToolBar;
import javafx.scene.layout.StackPane;
import javafx.stage.Stage;

public class ToolBarExample extends Application {

    @Override
    public void start(Stage primaryStage) {
        // 创建一个ToolBar
        ToolBar toolBar = new ToolBar();

        // 创建几个按钮并添加到ToolBar中
        Button button1 = new Button("Button 1");
        Button button2 = new Button("Button 2");
        Button button3 = new Button("Button 3");

        // 可以通过ToolBar的getItems()方法添加按钮，或者直接在ToolBar的构造函数中添加
        toolBar.getItems().addAll(button1, button2, button3);

        // 创建一个布局容器（这里使用StackPane只是为了简单起见，实际中可能使用其他更适合的容器）
        StackPane root = new StackPane();

        // 将ToolBar添加到布局容器中
        // 注意：StackPane默认会将子节点居中并堆叠显示，但ToolBar通常会根据内容自动调整其宽度和高度
        // 如果你想要ToolBar充满整个布局容器的宽度，可能需要使用不同的布局容器（如HBox或BorderPane）
        root.getChildren().add(toolBar);

        // 设置场景，并指定布局容器为根节点
        Scene scene = new Scene(root, 300, 100); // 场景大小可以根据需要调整

        // 设置舞台的标题和场景
        primaryStage.setTitle("ToolBar Example");
        primaryStage.setScene(scene);

        // 显示舞台
        primaryStage.show();
    }

    public static void main(String[] args) {
        launch(args); // 启动JavaFX应用程序
    }
}

  • 然后，我们创建了一个StackPane作为布局容器，并将ToolBar添加到其中。

注意

请注意，StackPane会将子节点居中并堆叠显示，这通常不是展示工具栏的最佳方式。在实际应用中，你可能想要使用HBox（水平布局容器）来水平排列工具栏中的控件，或者将ToolBar直接放置在BorderPane的顶部区域，以便在应用程序窗口的顶部显示工具栏。

# 示例二

package code.javafx.learn.JavaFX布局;

import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.control.ToolBar;
import javafx.scene.layout.BorderPane;
import javafx.scene.layout.StackPane;
import javafx.stage.Stage;

public class ToolBarExample extends Application {

    @Override
    public void start(Stage primaryStage) {
        // 创建一个ToolBar
        ToolBar toolBar = new ToolBar();

        // 创建几个按钮并添加到ToolBar中
        Button button1 = new Button("Button 1");
        Button button2 = new Button("Button 2");
        Button button3 = new Button("Button 3");

        // 可以通过ToolBar的getItems()方法添加按钮，或者直接在ToolBar的构造函数中添加
        toolBar.getItems().addAll(button1, button2, button3);

        // 创建一个布局容器（这里使用StackPane只是为了简单起见，实际中可能使用其他更适合的容器）
        BorderPane root = new BorderPane ();
        root.setTop(toolBar); // 将ToolBar添加到BorderPane的顶部

        // 将ToolBar添加到布局容器中
        // 注意：StackPane默认会将子节点居中并堆叠显示，但ToolBar通常会根据内容自动调整其宽度和高度
        // 如果你想要ToolBar充满整个布局容器的宽度，可能需要使用不同的布局容器（如HBox或BorderPane）
        // 设置场景，并指定布局容器为根节点
        Scene scene = new Scene(root, 600, 400); // 场景大小可以根据需要调整

        // 设置舞台的标题和场景
        primaryStage.setTitle("ToolBar Example");
        primaryStage.setScene(scene);

        // 显示舞台
        primaryStage.show();
    }

    public static void main(String[] args) {
        launch(args); // 启动JavaFX应用程序
    }
}
