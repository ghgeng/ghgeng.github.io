Java JDesktopPane

# JDesktopPane

JDesktopPane类可用于创建“多文档”应用程序。多文档应用程序可以包含多个窗口。我们通过将主窗口中的contentPane作为JDesktopPane类或子类的实例来实现。内部窗口将JInternalFrame的实例添加到JdesktopPane实例。内部窗口是JInternalFrame或其子类的实例。

# 示例代码1：

package code.swing.learn;

import javax.swing.*;
import java.awt.*;

/**
 * @author :ghgeng
 * @date 2023/7/20 17:05
 */
public class JDesktopPaneTest extends JFrame {

    JDesktopPaneTest() {
        // 获取当前容器面板
        Container contentPane = this.getContentPane();
        // 创建文档布局
        CustomDesktopTest customDesktopTest = new CustomDesktopTest();
        customDesktopTest.display();
        // 把当前文档布局添加到面板中
        contentPane.add(customDesktopTest, BorderLayout.CENTER);

        this.setSize(400, 400);
        this.setVisible(true);
        this.setLocationRelativeTo(null);
        this.setTitle("JDesktopPaneTest");
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    }


    class CustomDesktopTest extends JDesktopPane {
        // 创建三个窗口
        public void display() {
            int x = 30, y = 30;
            for (int a = 0; a <= 3; a++) {
                // 创建JInternalFrame
                JInternalFrame jInternalFrame = new JInternalFrame("JInternalFrame" + a,
                        true,
                        true,
                        true
                        , true);

                jInternalFrame.setBounds(x, y, 250, 85);
                // 在 jInternalFrame 容器中添加内容
                Container ji = jInternalFrame.getContentPane();
                ji.add(new JLabel("我爱中国"));
                // 添加到当前容器中
                this.add(jInternalFrame);
                jInternalFrame.setVisible(true);
                y += 30;
            }
        }
    }

    public static void main(String[] args) {
        new JDesktopPaneTest();
    }
}
