JavaFX Cubic Curve

提示

一般来说，三次曲线是3阶曲线。在JavaFX中，我们可以通过实例化JavaFX .scene.shape. cubiccurve类来创建三次曲线。该类包含表中定义的各种属性以及setter方法。为了创建所需的三次曲线，需要设置这些属性。

# 示例

三次曲线

package code.javafx;

import javafx.application.Application;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.effect.DropShadow;
import javafx.scene.paint.Color;
import javafx.scene.shape.CubicCurve;
import javafx.scene.shape.Polygon;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2023/9/20 17:39
 */
public class JavaFXCubicCurve extends Application {

    @Override
    public void start(Stage primaryStage) throws Exception {
        primaryStage.setTitle("JavaFXCubicCurve");
        Group group = new Group();
        CubicCurve c = new CubicCurve();
        c.setStartX(20);
        c.setStartY(100);
        c.setControlX1(300);
        c.setControlX2(200);
        c.setControlY1(100);
        c.setControlY2(90);
        c.setEndX(100);
        c.setEndY(300);

        c.setFill(Color.RED);
        c.setEffect(new DropShadow());

        group.getChildren().addAll(c);
        Scene scene = new Scene(group, 300, 400, Color.ANTIQUEWHITE);
        primaryStage.setScene(scene);
        primaryStage.show();
    }


    public static void main(String[] args) {
        launch(args);
    }
}
