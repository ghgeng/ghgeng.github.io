# DDL

# 添加新列：

ALTER TABLE IF EXISTS table_name ADD COLUMN  IF NOT EXISTS new_column_name column_type [not null] [DEFAULT 0|''];
1

# 删除列：

ALTER TABLE IF EXISTS table_name DROP COLUMN IF EXISTS column_name;
1

# 删除表

DROP TABLE IF EXISTS table_name ;
1

# 修改列类型或约束：

ALTER TABLE IF EXISTS table_name ALTER COLUMN column_name TYPE new_column_type;
1

# 重命名列：

ALTER TABLE IF EXISTS table_name RENAME COLUMN old_column_name TO new_column_name;
1

# 添加约束：

ALTER TABLE IF EXISTS table_name ADD CONSTRAINT constraint_name CHECK (column_name > 0);
1

# 删除约束：

ALTER TABLE IF EXISTS table_name DROP CONSTRAINT constraint_name;
1

# 重命名表：

ALTER TABLE IF EXISTS old_table_name RENAME TO new_table_name;
