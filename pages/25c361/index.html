Java JPanel

# JPanel

JPanel:表示一个容器，或者一个面板
JPanel是一个最简单的容器类。它提供了应用程序可以附加任何其他组件的空间。它继承了JComponents类。 它没有标题栏。

# 常用构造函数:

Constructor Description
JPanel() 它用于创建一个具有双缓冲区和流布局的新JPanel。
JPanel(boolean isDoubleBuffered) 创建一个新的 JPanel与 FlowLayout和指定的缓冲策略。
JPanel(LayoutManager layout) 使用指定的布局管理器创建一个新的缓冲JPanel。

# 示例代码1:

package code.swing.learn;

import javax.swing.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

/**
 * @author :ghgeng
 * @date 2023/6/3 9:40
 */
public class MySwing extends JFrame {

    private static JPanel root = new JPanel();

    public static void main(String[] args) {
        MySwing mySwing = new MySwing();
        mySwing.initMyFrame();
        mySwing._02();
    }

    public void initMyFrame() {
        this.setTitle("第一个窗口");
        // 当关闭窗口时 ，默认关闭程序
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        // 设置窗口(组件)大小
        this.setSize(400, 600);
        // 居中展示
        this.setLocationRelativeTo(null);
        // 设置面板，所有的组件都可以 .add() 放上去
        // 就好比有一个画板，画画之前放一张纸上去，大小可以自己设置
        this.setContentPane(root);
    }

    /**
     * 容器与控件
     * JPanel:表示一个容器，或者一个面板
     */
    public void _02() {
        // 显示窗口
        this.setVisible(true);
    }
}

# 示例代码2:

public class PanelExample extends JFrame {
    PanelExample() {
        this.setTitle("Panel Example");

        JPanel panel = new JPanel();
            panel.setBounds(40, 80, 200, 200);
            panel.setBackground(Color.gray);
        JButton b1 = new JButton("Button 1");
            b1.setBounds(50, 100, 80, 30);
            b1.setBackground(Color.yellow);
        JButton b2 = new JButton("Button 2");
            b2.setBounds(100, 100, 80, 30);
            b2.setBackground(Color.green);
        panel.add(b1);
        panel.add(b2);

        this.add(panel);
        this.setSize(400, 400);
        this.setLayout(null);
        this.setVisible(true);
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    }

    public static void main(String args[]) {
        new PanelExample();

    }
}
