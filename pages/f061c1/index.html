Java JComboBox

# JComboBox 下拉列表

选择类的对象用于显示弹出式选择菜单。用户选择的选项显示在菜单的顶部。它继承了JComponent类。

# 常用构造函数:

Constructor Description
JComboBox() 创建一个具有默认数据模型的JComboBox。
JComboBox(Object[] items) 创建一个包含指定数组中的元素的JComboBox。
JComboBox(Vector<?> items) 创建一个JComboBox，其中包含指定Vector中的元素。

# 常用方法

Methods Description
void addItem(Object anObject) 它用于向项列表中添加项。
void removeItem(Object anObject) 用于删除项目列表中的项目。
void removeAllItems() 它用于从列表中删除所有项。
void setEditable(boolean b) 它用于确定JComboBox是否可编辑。
void addActionListener(ActionListener a) 用于添加ActionListener。
void addItemListener(ItemListener i) 它用于添加ItemListener。

# 示例代码1：

package code.swing.learn;

import javax.swing.*;
import java.awt.*;

/**
 * @author :ghgeng
 * @date 2023/7/5 22:33
 */
public class JComboBoxTest {

    public static void main(String[] args) {
        JFrame jFrame = new JFrame();

        //创建下拉的数组
        Integer nums[] = new Integer[]{1,2,3,4,5000};
        JComboBox jComboBox = new JComboBox(nums);
        jComboBox.setBounds(100, 100, 2, 100);
        jFrame.add(jComboBox);

        jFrame.setLayout(new FlowLayout());
        jFrame.setLocationRelativeTo(null);
        jFrame.setVisible(true);
        jFrame.setSize(400, 400);
    }
}

# 示例代码2：

package code.swing.learn;

import javax.swing.*;
import java.awt.*;

/**
 * @author :ghgeng
 * @date 2023/7/5 22:33
 */
public class JComboBoxTest {

    public static void main(String[] args) {
        JFrame jFrame = new JFrame();

        JPanel jPanel = new JPanel();
        JComboBox<String> jComboBox = new JComboBox<>();
        jPanel.add(jComboBox);

        jComboBox.addItem("ONE");
        jComboBox.addItem("TWO");
        jComboBox.addItem("THREE");

        // 默认选中哪个选项，避免数组越界
        jComboBox.setSelectedIndex(1);
        // 也是默认选中哪个，不推荐
//        jComboBox.setSelectedItem("THREE");

        JButton jButton = new JButton("测试");
        jPanel.add(jButton);
        // 添加事件
        jButton.addActionListener((actionEvent) -> {
            int itemCount = jComboBox.getItemCount();
            System.out.println("下拉列表总数为：" + itemCount);

            String itemAt = jComboBox.getItemAt(0);
            System.out.println("选中的下拉为：" + itemAt);

            String selectedItem = (String) jComboBox.getSelectedItem();
            System.out.println("选中的数据为：" + selectedItem);

            int selectedIndex = jComboBox.getSelectedIndex();
            System.out.println("选中的下标为：" + selectedIndex);
            // 通过下标获取数据
            System.out.println("通过下标获取数据为：" + jComboBox.getItemAt(selectedIndex));
        });

        jFrame.add(jPanel);
        jFrame.setLayout(new FlowLayout());
        jFrame.setLocationRelativeTo(null);
        jFrame.setVisible(true);
        jFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        jFrame.setSize(400, 400);
    }
}
