Java JButton

# JButton组件

JButton:表示一个按钮控件
JButton类用于创建具有独立于平台实现的标记按钮。当按钮被按下时，应用程序会产生一些动作。它继承了AbstractButton类。

# 常用构造函数:

Constructor Description
JButton() 它创建了一个没有文本和图标的按钮。
JButton(String s) 它创建一个带有指定文本的按钮。
JButton(Icon i) 它用指定的图标对象创建一个按钮。

# 常用方法

Methods Description
void setText(String s) 用于在按钮上设置指定的文本。
String getText() 它用于返回按钮的文本。
void setEnabled(boolean b) 用于启用或禁用按钮。
void setIcon(Icon b) 用于设置按钮上指定的图标。
Icon getIcon() 它用于获取按钮的图标。
void setMnemonic(int a) 它用于设置按钮上的助记符。
void addActionListener(ActionListener a) 它用于向此对象添加操作侦听器。

package code.swing.learn;

import javax.swing.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
/**
 * @author :ghgeng
 * @date 2023/6/3 9:40
 */
public class MySwing extends JFrame {

    private static JPanel root = new JPanel();

    public static void main(String[] args) {
        MySwing mySwing = new MySwing();
        mySwing.initMyFrame();
        mySwing._02();
    }
    public void initMyFrame() {
        this.setTitle("第一个窗口");
        // 当关闭窗口时 ，默认关闭程序
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        // 设置窗口(组件)大小
        this.setSize(400, 600);
        // 居中展示
        this.setLocationRelativeTo(null);
        // 设置面板，所有的组件都可以 .add() 放上去
        // 就好比有一个画板，画画之前放一张纸上去，大小可以自己设置
        this.setContentPane(root);
    }
    /**
     * 容器与控件
     * JPanel:表示一个容器，或者一个面板
     * JButton:表示一个按钮控件
     */
    public void _02() {
        JButton jButton = new JButton("我是一个按钮");
        root.add(jButton);
        // 显示窗口
        this.setVisible(true);
    }
}

# 带有ActionListener的Java JButton示例

  • 代码:
package code.swing.learn;

import javax.swing.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

/**
 * @author :ghgeng
 * @date 2023/6/3 9:40
 */
public class MySwing extends JFrame {

    private static JPanel root = new JPanel();

    public static void main(String[] args) {
        MySwing mySwing = new MySwing();
        mySwing.initMyFrame();
        mySwing._04();
    }

    public void initMyFrame() {
        this.setTitle("第一个窗口");
        // 当关闭窗口时 ，默认关闭程序
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        // 设置窗口(组件)大小
        this.setSize(400, 600);
        // 居中展示
        this.setLocationRelativeTo(null);
        // 设置面板，所有的组件都可以 .add() 放上去
        // 就好比有一个画板，画画之前放一张纸上去，大小可以自己设置
        this.setContentPane(root);
    }
        /**
         * 按钮事件处理
         * 调用 JButton 按钮的 addActionListener() 方法
         */
        public void _04() {
            // 创建一个按钮组件放到面板中
            JButton jButton = new JButton("测试");
            root.add(jButton);
            // 添加时间到Lable中去
            JLabel jLabel = new JLabel("00:00:00");
            root.add(jLabel);
    
            // 第一种写法，入参为ActionListener 类型，创建一个内类实现ActionListener
            MyActionListener myActionListener = new MyActionListener();
            jButton.addActionListener(myActionListener);
    
            // 第二种写法, 直接 new ActionListener ，用匿名内部类实现
            jButton.addActionListener(new ActionListener() {
                @Override
                public void actionPerformed(ActionEvent e) {
                    System.out.println("第二种写法");
                }
            });
            // 第三种写法 , 内明内部类简化成 （e） -> {} , 括号可以省略
            jButton.addActionListener(e -> {
                System.out.println("第三种写法");
                SimpleDateFormat sf = new SimpleDateFormat();
                String time = sf.format(new Date());
                jLabel.setText(time);
                System.out.println(time);
            });
            // 显示窗口
            this.setVisible(true);
        }
    private class MyActionListener implements ActionListener {
        @Override
        public void actionPerformed(ActionEvent e) {
            System.out.println("第一种写法");
        }
    }
}
