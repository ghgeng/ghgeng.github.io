JavaFX ScrollBar

# JavaFX ScrollBar

提示

JavaFX滚动条用于向用户提供滚动条，以便用户可以向下滚动应用程序页面。可以通过实例化javafx.scene.control.ScrollBar类来创建。 以下代码在我们的应用程序中实现了scrollbar。

# 示例一


package code.javafx.JavaFXUI;

import javafx.application.Application;
import javafx.beans.value.ChangeListener;
import javafx.beans.value.ObservableValue;
import javafx.geometry.Orientation;
import javafx.scene.Scene;
import javafx.scene.control.ScrollBar;
import javafx.scene.layout.StackPane;
import javafx.stage.Stage;

public class ScrollBarTest extends Application {

    @Override
    public void start(Stage primaryStage) throws Exception {
        // TODO Auto-generated method stub
        ScrollBar s = new ScrollBar();

        s.valueProperty().addListener(new ChangeListener() {
            @Override
            public void changed(ObservableValue observable, Object oldValue, Object newValue) {
                System.out.println(newValue);
            }
        });

        StackPane root = new StackPane();
        root.getChildren().add(s);
        Scene scene = new Scene(root, 300, 200);
        primaryStage.setScene(scene);
        primaryStage.setTitle("ScrollBar Example");
        primaryStage.show();

    }
    public static void main(String[] args) {
        launch(args);
    }
}

# 示例二

滚动条既可以水平显示，也可以垂直显示。在JavaFX中，我们可以设置滚动条的任何方向。setOrientation()并传递方向。属性作为参数添加到方法中。 鼠标滚轮设置大小

  • setMin()
  • setMax()
  • setValue()


package code.javafx.JavaFXUI;

import javafx.application.Application;
import javafx.beans.value.ChangeListener;
import javafx.beans.value.ObservableValue;
import javafx.geometry.Orientation;
import javafx.scene.Scene;
import javafx.scene.control.ScrollBar;
import javafx.scene.layout.StackPane;
import javafx.stage.Stage;

public class ScrollBarTest extends Application {

    @Override
    public void start(Stage primaryStage) throws Exception {
        // TODO Auto-generated method stub
        ScrollBar s = new ScrollBar();
        s.setMin(0);
        s.setMax(200);
        // 设置初始值大小
        s.setValue(50);
        // 鼠标滚轮设置的增量值
        s.setUnitIncrement(5);
        s.setBlockIncrement(2);
        // 垂直显示
        s.setOrientation(Orientation.VERTICAL);

        s.valueProperty().addListener(new ChangeListener() {
            @Override
            public void changed(ObservableValue observable, Object oldValue, Object newValue) {
                System.out.println(newValue);
            }
        });

        StackPane root = new StackPane();
        root.getChildren().add(s);
        Scene scene = new Scene(root, 300, 200);
        primaryStage.setScene(scene);
        primaryStage.setTitle("ScrollBar Example");
        primaryStage.show();

    }

    public static void main(String[] args) {
        launch(args);
    }
}
