JavaFX StackPane

# StackPane

提示

在JavaFX中，StackPane是一个布局容器，它按照Z轴（即堆叠顺序）来排列其子节点。子节点会被堆叠在一起，并且后添加的节点会覆盖在先前添加的节点之上。以下是一个简单的StackPane代码示例，展示了如何使用它以及如何在其中添加和排列子节点。

# 示例代码

package code.javafx.learn.JavaFX布局;

import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.control.Label;
import javafx.scene.layout.StackPane;
import javafx.stage.Stage;

public class StackPaneExample extends Application {

    @Override
    public void start(Stage primaryStage) {
        // 创建一个StackPane
        StackPane stackPane = new StackPane();

        // 创建一些子节点
        Label label = new Label("Hello, StackPane!");
        label.setStyle("-fx-background-color: lightblue;"); // 给标签设置背景颜色以更好地显示它

        Button button = new Button("Click Me!");
        button.setOnAction(e -> System.out.println("Button clicked!")); // 设置按钮的点击事件

        // 将子节点添加到StackPane中
        // 注意：后添加的节点会覆盖在先前添加的节点之上
        stackPane.getChildren().add(label); // 首先添加标签
        stackPane.getChildren().add(button); // 然后添加按钮，这会覆盖标签的一部分（或全部，取决于按钮的大小和位置）

        // 设置场景和舞台
        Scene scene = new Scene(stackPane, 300, 200); // 场景大小为300x200

        primaryStage.setTitle("StackPane Example"); // 舞台标题
        primaryStage.setScene(scene); // 将场景设置给舞台
        primaryStage.show(); // 显示舞台
    }

    public static void main(String[] args) {
        launch(args); // 启动JavaFX应用程序
    }
}
