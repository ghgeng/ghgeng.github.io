JavaFX 光点类

提示

JavaFX Light.Spot
这种效果通过一个点光源照亮节点。聚光光源是光在各个方向上都衰减的光源。光源的强度取决于光源与节点之间的距离。类javafx.scene.effect.Light.Spot表示这种效果。我们只需要实例化这个类来在节点上生成适当的光。

# 属性

方法 描述
setPointsAtX(double value) 这是一个双类型属性。它表示光的方向向量的X坐标
setPointsAtY(double value) 这是一个双类型属性。它表示光的方向向量的Y坐标
setPointsAtZ(double value) 这是一个双类型属性。它表示光的方向矢量的Z坐标
setSpecularExponent(double value) 这是一个双类型属性。它代表投机指数。这是用来改变光源的焦点

# 示例1

package code.javafx.JavaFXEffects.JavaFXLight;

import javafx.application.Application;
import javafx.geometry.VPos;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.effect.Light;
import javafx.scene.effect.Lighting;
import javafx.scene.paint.Color;
import javafx.scene.shape.Circle;
import javafx.scene.text.Font;
import javafx.scene.text.FontWeight;
import javafx.scene.text.Text;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2024/1/29 16:53
 */
public class Spot extends Application {

    @Override
    public void start(Stage primaryStage) throws Exception {
        primaryStage.setTitle("JavaLighting.Spot Example");
        Text text = new Text();
        text.setFont(Font.font(null, FontWeight.BOLD, 40));
        text.setX(60);
        text.setY(50);
        text.setText("Welcome to BeiJing");
        text.setFill(Color.RED);

        Circle circle = new Circle();
        circle.setCenterX(300.0f);
        circle.setCenterY(160.0f);
        circle.setRadius(100.0f);
        circle.setFill(Color.YELLOW);

        Light.Spot light = new Light.Spot();
        light.setColor(Color.WHITE);
        light.setX(150);
        light.setY(0);
        light.setZ(45);

        Lighting lighting = new Lighting(light);

        text.setEffect(lighting);
        circle.setEffect(lighting);

        Group root = new Group(text, circle);
        Scene scene = new Scene(root, 600, 300);
        primaryStage.setScene(scene);

        primaryStage.show();

    }

    public static void main(String[] args) {
        launch(args);
    }
}
