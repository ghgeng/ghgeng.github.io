Java JTabbedPane

# JTabbedPane

JTabbedPane类用于通过单击带有给定标题或图标的选项卡来在一组组件之间切换。它继承了JComponent类。

# 常用构造函数：

Constructor Description
JTabbedPane() 创建一个空的TabbedPane，其默认选项卡位置为JTabbedPane.Top。
JTabbedPane(int tabPlacement) 创建具有指定选项卡位置的空TabbedPane。
JTabbedPane(int tabPlacement, int tabLayoutPolicy) 使用指定的选项卡位置和选项卡布局策略创建空TabbedPane。

# 示例代码1：

package code.swing.learn;

import javax.swing.*;

/**
 * @author :ghgeng
 * @date 2023/7/17 16:38
 */
public class JTabbedPaneTest {

    public static void main(String[] args) {
        JFrame j = new JFrame("JScrollBarTest");
        // 创建三个窗口
        JPanel jPanel1 = new JPanel();
        JPanel jPanel2 = new JPanel();
        JPanel jPanel3 = new JPanel();

        JTextArea jTextArea = new JTextArea(200,200);
        jPanel1.add(jTextArea);

        // 创建JTabbedPane
        JTabbedPane jTabbedPane = new JTabbedPane();
        jTabbedPane.setBounds(20, 20, 200, 200);
        jTabbedPane.add("main", jPanel1);
        jTabbedPane.add("helper", jPanel2);
        jTabbedPane.add("other", jPanel3);

        j.add(jTabbedPane);
        // 显示窗口
        j.setSize(400, 400);
        j.setLayout(null);
        j.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        j.setLocationRelativeTo(null);
        j.setVisible(true);
    }
}
