Java JList

# JList

JList类的对象表示文本项的列表。可以设置文本项列表，以便用户可以选择一个项或多个项。它继承了JComponent类。

# 常用构造器:

Constructor Description
JList() 用一个空的只读的模型创建一个JList。
JList(ary[] listData) 创建一个JList,它显示指定数组中的元素。
JList(ListModel<ary> dataModel) 创建一个JList,它显示指定的、非null模型的元素。

# 常用方法:

Methods Description
Void addListSelectionListener(ListSelectionListener listener) 它用于在列表中添加一个侦听器,每次更改到选择时都要被通知。
int getSelectedIndex() 它用于返回最小的选定单元格索引。
void setListData(Object[] listData) 它用于创建来自一个对象数组的只读列表模型。

# 示例代码1：

package code.swing.learn;

import javax.swing.*;

/**
 * @author :ghgeng
 * @date 2023/7/11 22:35
 */
public class JListText {

    public static void main(String[] args) {
        JFrame jFrame = new JFrame();
        // 创建默认list列表
        DefaultListModel defaultListModel = new DefaultListModel();
        defaultListModel.addElement("item1");
        defaultListModel.addElement("item2");
        defaultListModel.addElement("item3");
        defaultListModel.addElement("item4");

        JList<String> jList = new JList(defaultListModel);
        jList.setBounds(50, 50, 100, 100);
        jFrame.add(jList);

        jFrame.setLayout(null);
        jFrame.setSize(400, 400);
        jFrame.setLocationRelativeTo(null);
        jFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        jFrame.setVisible(true);
    }
}

# 示例代码2：

package code.swing.learn;

import javax.swing.*;

/**
 * @author :ghgeng
 * @date 2023/7/11 22:35
 */
public class JListText {

    public static void main(String[] args) {
        JFrame jFrame = new JFrame();
        JLabel jLabel = new JLabel("");
        jLabel.setBounds(100, 200, 300, 30);
        jFrame.add(jLabel);

        // 创建默认list列表
        DefaultListModel defaultListModel = new DefaultListModel();
        defaultListModel.addElement("Java");
        defaultListModel.addElement("Python");
        defaultListModel.addElement("C++");
        defaultListModel.addElement("Html");

        JList<String> jList = new JList(defaultListModel);
        jList.setBounds(50, 50, 50, 100);
        jFrame.add(jList);

        DefaultListModel defaultListModel2 = new DefaultListModel();
        defaultListModel2.addElement("基础");
        defaultListModel2.addElement("入门");
        defaultListModel2.addElement("中级");
        defaultListModel2.addElement("高级");

        JList<String> jList2 = new JList(defaultListModel2);
        jList2.setBounds(150, 50, 50, 100);
        jFrame.add(jList2);

        JButton jButton = new JButton("show");
        jButton.setBounds(100, 150, 100, 30);
        jFrame.add(jButton);

        // 给按钮添加事件
        jButton.addActionListener(e -> {
            String data = "";

            data = jList.getSelectedValue()+ "-";

            data += jList2.getSelectedValue();

            System.out.println(data);
            jLabel.setText(data);
        });

        jFrame.setLayout(null);
        jFrame.setSize(400, 400);
        jFrame.setLocationRelativeTo(null);
        jFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        jFrame.setVisible(true);
    }
}
