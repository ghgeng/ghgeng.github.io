# 增加多系统的兼容代码

善窗口行为(对应官网的管理窗口生命周期)
windows所有的窗口都关闭了，应用就自动退出
mac退出应用需要快捷键command+q,不然是关闭窗口command+w

  • 代码重构如下:
// 这里也是关于ready的另一种写法
app.whenReady().then(() => {
        createWindow()
        // 管理窗口的生命周期   激活窗口
        app.on('activate', () => {
            if (BrowserWindow.getAllWindows().length === 0) createWindow()
        })
    }
)

function createWindow() {
const win = new BrowserWindow({
        width: 500,//窗口宽度
        height: 500,//窗口高度
        autoHideMenuBar: true,//自动隐藏菜单档
        alwaysOnTop: true,//置顶
        x: 0,//窗口位置x坐标
        y: 0//窗口位置y坐标
    })
    //加载一个本地页面
    win.loadFile('./pages/index.html')
}


// 管理窗口的生命周期   退出
app.on('window-all-closed', () => {
    if (process.platform !== 'darwin') app.quit()
})

# 完整代码

  • # main.js

const {app, BrowserWindow, ipcMain} = require('electron')
const path = require('path')
const fs = require('fs')


// 这里也是关于ready的另一种写法
app.whenReady().then(() => {
        createWindow()
        // 管理窗口的生命周期   激活窗口
        app.on('activate', () => {
            if (BrowserWindow.getAllWindows().length === 0) createWindow()
        })
    }
)


function savefile(event, data) {
    console.log(data)
    fs.writeFileSync('hello.txt', data)
}

function readfile() {
   return  fs.readFileSync('hello.txt').toString()
}

function createWindow() {
    const win = new BrowserWindow({
        width: 500,//窗口宽度
        height: 600,//窗口高度
        autoHideMenuBar: true,//自动隐藏菜单档
        alwaysOnTop: false,//置顶
        x: 0,//窗口位置x坐标
        y: 0,//窗口位置y坐标
        webPreferences: {
            preload: path.resolve(__dirname, './preload.js')
        }
    })
    win.loadFile('./pages/index.html')
    ipcMain.on('save-file', savefile)
    ipcMain.handle('read-file', readfile)
    win.openDevTools()  //自动打开调试窗口

}

// 管理窗口的生命周期   退出
app.on('window-all-closed', () => {
    if (process.platform !== 'darwin') app.quit()
})

# 打包

  • 使用electron-builder打包
npm install electron-builder -D
1
  • package.json配置说明
{
  "name": "eclectron",
  "version": "1.0.0",
  "main": "main.js", // 应用程序入口文件
  "scripts": {
    "start": "electron .",  // 使用electron 命令启动
    "build": "electron-builder" // 使用 electron-builder 打包
  },
  "build": {
    "appId": "com.ghgeng.demo", //应用程序唯一标识
    // 打包windows平台安装包具体位置
    "win": {
      "icon":"./logo.ico", // 应用图标
      "target": [
        {
          "target": "nsis", // 使用NSIS(mis) 作为安装程序格式 
          "arch": [ 
            "x64" // 生成64为安装包
          ]
        }
      ]
    },
    "nsis": {
      "oneClick": false, // 设置为false 使安装程序显示安装向导界面，而不是一键安装
      "perMachine": true, // 允许每台机器安装一次，而不是每个用户都安装
      "allowToChangeInstallationDirectory": true // 允许用户在安装过程中选择安装目录
    }
  },
  "author": "ghgeng",
  "license": "ISC", // 许可证信息
  "description": "electron demo", // 描述
  "devDependencies": { // 环境依赖
    "electron": "^31.3.1",
    "electron-builder": "^24.13.3",
    "nodemon": "^3.1.4"
  }
}
