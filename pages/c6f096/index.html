JavaFX AnchorPane

# AnchorPane

提示

在JavaFX中，AnchorPane是一个布局容器，它允许你通过锚点（即边缘）来定位其子节点。这意呀着你可以指定子节点相对于AnchorPane的顶部、右侧、底部和左侧边缘的距离。以下是一个简单的AnchorPane代码示例，展示了如何在一个AnchorPane中布局两个按钮，并设置它们的锚点位置。

# 示例代码

package code.javafx.learn.JavaFX布局;

import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.layout.AnchorPane;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2024/9/4 14:23
 */

public class AnchorPaneTest extends Application {

    @Override
    public void start(Stage primaryStage) {
        // 创建一个AnchorPane实例
        AnchorPane anchorPane = new AnchorPane();

        // 创建两个按钮
        Button btnTopLeft = new Button("左上角");
        Button btnBottomRight = new Button("右下角");

        // 将按钮添加到AnchorPane中
        anchorPane.getChildren().addAll(btnTopLeft, btnBottomRight);

        // 设置按钮的锚点位置
        // 注意：这里设置的是按钮与AnchorPane边缘的距离
        AnchorPane.setTopAnchor(btnTopLeft, 20.0); // 距离顶部20
        AnchorPane.setLeftAnchor(btnTopLeft, 20.0); // 距离左侧20

        AnchorPane.setBottomAnchor(btnBottomRight, 20.0); // 距离底部20
        AnchorPane.setRightAnchor(btnBottomRight, 20.0); // 距离右侧20

        // 设置场景，并将AnchorPane添加到场景中
        Scene scene = new Scene(anchorPane, 300, 250);

        // 配置并显示舞台
        primaryStage.setTitle("AnchorPane 示例");
        primaryStage.setScene(scene);
        primaryStage.show();
    }

    public static void main(String[] args) {
        launch(args);
    }

}
