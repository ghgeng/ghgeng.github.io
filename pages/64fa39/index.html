JavaFX ScrollPane

# ScrollPane

提示

这个示例创建了一个简单的ScrollPane，其中包含一个VBox（垂直布局容器），VBox里又包含了一些Button（按钮）。由于我们故意让VBox的高度大于ScrollPane的视图区域，因此滚动条会自动出现，允许用户滚动查看所有按钮。

# 示例代码

package code.javafx.learn.JavaFX布局;

import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.control.ScrollPane;
import javafx.scene.layout.VBox;
import javafx.stage.Stage;

public class ScrollPaneExample extends Application {

    @Override
    public void start(Stage primaryStage) {
        // 创建一个VBox，它将包含我们的按钮
        VBox vBox = new VBox();
        vBox.setSpacing(10); // 设置VBox中子节点之间的间距

        // 向VBox中添加一些按钮
        for (int i = 0; i < 20; i++) { // 假设我们添加20个按钮
            Button button = new Button("Button " + (i + 1));
            vBox.getChildren().add(button); // 将按钮添加到VBox中
        }

        // 创建一个ScrollPane，并将VBox设置为其内容
        ScrollPane scrollPane = new ScrollPane(vBox);
        scrollPane.setFitToWidth(true); // 使ScrollPane的宽度适应其内容的宽度（可选）
        scrollPane.setVbarPolicy(ScrollPane.ScrollBarPolicy.ALWAYS); // 总是显示垂直滚动条（可选，默认为AUTO）

        // 设置场景和舞台
        Scene scene = new Scene(scrollPane, 300, 200); // 场景大小为300x200

        primaryStage.setTitle("ScrollPane Example"); // 舞台标题
        primaryStage.setScene(scene); // 将场景设置给舞台
        primaryStage.show(); // 显示舞台
    }

    public static void main(String[] args) {
        launch(args); // 启动JavaFX应用程序
    }
}
