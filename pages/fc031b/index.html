JavaFX TableView

提示

在JavaFX中，TableView是一个非常强大的控件，用于展示和操作表格数据。以下是一个简单的JavaFX TableView示例，并填充了一些示例数据。

我们创建了一个TableView实例，并为其添加了三个TableColumn实例，每个列都通过PropertyValueFactory来指定它应该显示Person对象的哪个属性。
我们将一些Person对象添加到TableView的数据列表中，并将其显示在舞台上。

# 示例代码

package code.test;

import javafx.application.Application;
import javafx.collections.FXCollections;
import javafx.collections.ObservableList;
import javafx.scene.Scene;
import javafx.scene.control.TableColumn;
import javafx.scene.control.TableView;
import javafx.scene.control.cell.PropertyValueFactory;
import javafx.scene.layout.StackPane;
import javafx.stage.Stage;
import lombok.Data;

public class TableViewExample extends Application {

    @Override
    public void start(Stage primaryStage) {

        ObservableList<Person> data = FXCollections.observableArrayList(
                new Person("John", "Doe"),
                new Person("John", "asdasda")
        );


        TableView<Person> tableView = new TableView<>();

        TableColumn<Person, String> firstNameCol = new TableColumn<>("First Name");
        firstNameCol.setCellValueFactory(new PropertyValueFactory<>("firstName"));

        TableColumn<Person, String> lastNameCol = new TableColumn<>("Last Name");
        lastNameCol.setCellValueFactory(new PropertyValueFactory<>("lastName"));

        tableView.getColumns().addAll(firstNameCol, lastNameCol);

        // 添加数据
        tableView.setItems(data);

        StackPane root = new StackPane();
        root.getChildren().add(tableView);

        Scene scene = new Scene(root, 300, 250);

        primaryStage.setTitle("TableView Example");
        primaryStage.setScene(scene);
        primaryStage.show();
    }

    @Data
    public static class Person {
        private final String firstName;
        private final String lastName;

        private Person(String fName, String lName) {
            this.firstName = fName;
            this.lastName = lName;
        }
    }

    public static void main(String[] args) {
        launch(args);
    }
}

# 代码示例二

  • javafx.beans.property 是JavaFX库中的一个包，它提供了用于封装Java对象的属性和其值变化的类。这些属性可以被绑定到UI元素上，当属性值改变时，UI会自动更新。
    • BooleanProperty：封装布尔类型的属性。
    • DoubleProperty：封装双精度浮点数类型的属性。
    • IntegerProperty：封装整型数值的属性。
    • ObjectProperty：封装任意对象类型的属性。
    • StringProperty：封装字符串类型的属性。
  • TableView强制渲染，可以调用refresh()方法。
属性
get() 获取属性的值
set(T value) 设置属性的值
getValue() 与get()方法相同，获取属性的值
setValue(T value) 与set(T value)方法相同，设置属性的值
addListener(ChangeListener<? super T> listener) 添加一个监听器来监听属性值的变化

package code.javafx.learn.控件;

import javafx.application.Application;
import javafx.beans.property.IntegerProperty;
import javafx.beans.property.SimpleIntegerProperty;
import javafx.beans.property.SimpleStringProperty;
import javafx.beans.property.StringProperty;
import javafx.collections.FXCollections;
import javafx.collections.ObservableList;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.control.TableColumn;
import javafx.scene.control.TableView;
import javafx.scene.control.cell.PropertyValueFactory;
import javafx.scene.layout.StackPane;
import javafx.stage.Stage;
import lombok.Data;

import java.util.Random;

public class TableViewWatchExample extends Application {

    public static void main(String[] args) {
        launch(args);
    }


    @Override
    public void start(Stage primaryStage) {
        // 创建数据
        ObservableList<Person> data = FXCollections.observableArrayList(
                new Person(1, "Alice", "alice@example.com"),
                new Person(2, "Bob", "bob@example.com"),
                new Person(3, "Charlie", "charlie@example.com")
        );

        // 创建TableView
        TableView<Person> tableView = new TableView<>();

        // 创建表格列
        TableColumn<Person, Integer> idCol = new TableColumn<>("ID");
        idCol.setCellValueFactory(new PropertyValueFactory<>("id"));

        TableColumn<Person, String> nameCol = new TableColumn<>("Name");
        nameCol.setCellValueFactory(new PropertyValueFactory<>("name"));

        TableColumn<Person, String> emailCol = new TableColumn<>("Email");
        emailCol.setCellValueFactory(new PropertyValueFactory<>("email"));

        // 将列添加到TableView
        tableView.getColumns().addAll(idCol, nameCol, emailCol);

        // 设置TableView的数据
        tableView.setItems(data);

        // 修改按钮
        Button button = new Button("测试修改监听");
        button.setOnMouseClicked(e -> {
            Random random = new Random();
            int randomInt = random.nextInt(20);
            Person person = data.get(0);
            person.setId(randomInt);

            // 重新设置TableView的数据
            tableView.refresh();
            // 选中一行
            Person selectedItem = tableView.getSelectionModel().getSelectedItem();
            if (selectedItem != null) {
                System.out.println(selectedItem.toString());
            }
        });

        // 设置场景和舞台
        StackPane root = new StackPane();

        root.getChildren().addAll(tableView, button);
        Scene scene = new Scene(root, 300, 250);

        primaryStage.setTitle("TableView Example");
        primaryStage.setScene(scene);
        primaryStage.show();


    }

    public static class Person {
        private final IntegerProperty id;
        private final StringProperty name;
        private final StringProperty email;

        public Person(Integer id, String name, String email) {
            this.id = new SimpleIntegerProperty(id);
            this.name = new SimpleStringProperty(name);
            this.email = new SimpleStringProperty(email);
            // 监听
            watch();
        }

        @Override
        public String toString() {
            return "Person{" +
                    "id=" + id.get() +
                    ", name=" + name.get() +
                    ", email=" + email.getValue() +
                    '}';
        }

        public void watch() {
            id.addListener((observable, oldValue, newValue) -> {
                System.out.println("Id changed from " + oldValue + " to " + newValue);
            });

            name.addListener((observable, oldValue, newValue) -> {
                System.out.println("Name changed from " + oldValue + " to " + newValue);
            });

            email.addListener((observable, oldValue, newValue) -> {
                System.out.println("Email changed from " + oldValue + " to " + newValue);
            });

        }

        public final int getId() {
            return id.get();
        }

        public final void setId(int mid) {
            id.setValue(mid);
        }


        public final String getName() {
            return name.get();
        }

        public void setName(String mname) {
            name.setValue(mname);
        }

        public final String getEmail() {
            return email.get();
        }
    }

}
