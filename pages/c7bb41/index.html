JavaFX Ellipse

# JavaFX椭圆

# 示例1

创建椭圆

package code.javafx;

import javafx.application.Application;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.paint.Color;
import javafx.scene.shape.Ellipse;
import javafx.scene.shape.Rectangle;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2023/9/20 17:39
 */
public class JavaFXEllipse extends Application {

    @Override
    public void start(Stage primaryStage) throws Exception {
        primaryStage.setTitle("JavaFXEllipse Example");
        Group group = new Group();
        // 创建椭圆
        Ellipse elipse = new Ellipse();
        elipse.setCenterX(100);
        elipse.setCenterY(100);
        elipse.setRadiusX(50);
        elipse.setRadiusY(80);
        group.getChildren().add(elipse);

        Scene scene = new Scene(group,200,300, Color.GRAY);
        primaryStage.setScene(scene);
        primaryStage.show();
    }


    public static void main(String[] args) {
        launch(args);
    }
}
