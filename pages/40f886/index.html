跳过SSlL验证

# 跳过SSlL验证

package com.isoftstone.phonepay.common.helper;

import java.io.BufferedReader;
import java.io.ByteArrayInputStream;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.io.OutputStream;
import java.net.HttpURLConnection;
import java.net.URL;
import java.net.URLEncoder;
import java.security.SecureRandom;
import java.util.Iterator;
import java.util.Map;

import javax.net.ssl.HttpsURLConnection;
import javax.net.ssl.KeyManager;
import javax.net.ssl.SSLContext;
import javax.net.ssl.TrustManager;

import org.apache.commons.io.IOUtils;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import com.isoftstone.phonepay.common.exception.AppException;
import com.isoftstone.phonepay.common.https.DefaultTrustManager;
import com.isoftstone.phonepay.common.https.TrustAnyHostnameVerifier;

/**
 * HTTP帮助类 
 * @author JITAO
 */
public class HttpHelper {
	
	/**
	 * logger
	 */
	private static final Logger logger = LoggerFactory.getLogger(HttpHelper.class);
	
	/**
	 * GET方法
	 */
	public static String METHOD_GET = "GET";
	
	/**
	 * POST方法
	 */
	public static String METHOD_POST = "POST";
	
	/**
	 * 请求http/https
	 * @param url URL
	 * @param encoding 请求及响应编码
	 * @param method 请求方法类型
	 * @param timeout 超时时间，单位：毫秒
	 * @param paramMap 参数MAP
	 * @param in 参数以数据流形式发送
	 * @return 响应返回内容
	 */
	private static String requestHttp(String url, String encoding, String method, int timeout, Map<String, String> paramMap, InputStream in) throws Exception {
		
		// 拼装发送数据
		StringBuilder sendData = new StringBuilder();
		if (paramMap != null && paramMap.size() > 0) {
			Iterator<String> keyIter = paramMap.keySet().iterator();
			while (keyIter.hasNext()) {
				String key = keyIter.next();
				String value = paramMap.get(key);
				if (sendData.length() > 0) {
					sendData.append("&");
				}
				sendData.append(key).append("=");
				if (value != null && value.length() > 0) {
					sendData.append(URLEncoder.encode(value, encoding));
				}
			}
		}
		
		// GET请求参数拼接至URL
		if (METHOD_GET.equals(method) && sendData.length() > 0) {
			if (url.contains("?")) {
				url = url + "&" + sendData.toString();
			} else {
				url = url + "?" + sendData.toString();
			}
		}
		
		// 创建URL对象
		URL urlObj = new URL(url);
		
		// 创建连接
		HttpURLConnection conn = (HttpURLConnection) urlObj.openConnection();;
		
		// https协议设置信任所有证书
		if (urlObj.getProtocol().equalsIgnoreCase("https")) {
			logger.debug("使用https协议，信任所有。");
			
			SSLContext ctx = SSLContext.getInstance("TLS");
			ctx.init(new KeyManager[0], new TrustManager[]{new DefaultTrustManager()}, new SecureRandom());
			
			((HttpsURLConnection) conn).setSSLSocketFactory(ctx.getSocketFactory());
			((HttpsURLConnection) conn).setHostnameVerifier(new TrustAnyHostnameVerifier());
		}
		
		// 设置连接和读取的超时时间
		conn.setConnectTimeout(timeout);
		conn.setReadTimeout(timeout);
		
		logger.debug("开始发送{}的{}请求。", url, method);
		if (METHOD_POST.equals(method)) {
			if (logger.isDebugEnabled()) {
				if (in != null) {
					logger.debug("POST请求参数通过数据流格式发送");
				} else {
					logger.debug("POST请求参数：{}", sendData.toString());
				}
			}
		}
		
		// 发送数据
		if (METHOD_GET.equals(method)) {
			conn.setUseCaches(false);
			conn.setRequestMethod("GET");
			conn.connect();
		} else {
			conn.setUseCaches(false);
			conn.setDoInput(true);
			conn.setDoOutput(true);
			conn.setRequestMethod("POST");
			conn.setRequestProperty("content-type", "application/x-www-form-urlencoded");
			
			OutputStream output = null;
			try {
				output = conn.getOutputStream();
				if (in != null) {
					IOUtils.copy(in, output);
				} else {
					output.write(sendData.toString().getBytes());
				}
				output.flush();
			} finally {
				if (in != null) {
					in.close();
				}
				if (output != null) {
					output.close();
				}
			}
		}
		
		// 读取回应结果
		StringBuilder returnContent = new StringBuilder();
		if (conn.getResponseCode() == HttpURLConnection.HTTP_OK) {
			InputStream input = null;
			try {
				input = conn.getInputStream();
				
				BufferedReader reader = new BufferedReader(new InputStreamReader(input, encoding));
				char[] chars = new char[128];
				int readCount = 0;
				while ((readCount = reader.read(chars)) != -1) {
					returnContent.append(chars, 0, readCount);
				}
			} finally {
				if (input != null) {
					input.close();
				}
			}
		} else {
			conn.disconnect();
			logger.debug("地址{}返回错误码：{}", url, conn.getResponseCode());
			throw new AppException("响应错误码：" + conn.getResponseCode());
		}
		conn.disconnect();
		String rc = returnContent.toString();
		
		logger.debug("请求{}成功，返回内容：{}", url, rc);
		return rc;
	}
	
	/**
	 * 使用key-value参数格式请求http/https
	 * @param url URL
	 * @param encoding 请求及响应编码
	 * @param method 请求方法类型
	 * @param timeout 超时时间，单位：毫秒
	 * @param paramMap 参数MAP
	 * @return 响应返回内容
	 */
	public static String requestHttpByParameter(String url, String encoding, String method, int timeout, Map<String, String> paramMap) throws Exception {
		return requestHttp(url, encoding, method, timeout, paramMap, null);
	}
	
	/**
	 * 使用数据流参数格式，POST方法，请求http/https
	 * @param url URL
	 * @param encoding 请求及响应编码
	 * @param timeout 超时时间，单位：毫秒
	 * @param in 参数以数据流形式发送
	 * @return 响应返回内容
	 */
	public static String requestPostByStream(String url, String encoding, int timeout, InputStream in) throws Exception {
		return requestHttp(url, encoding, HttpHelper.METHOD_POST, timeout, null, in);
	}
	
	/**
	 * 使用数据流参数格式，POST方法，请求http/https
	 * @param url URL
	 * @param encoding 请求及响应编码
	 * @param timeout 超时时间，单位：毫秒
	 * @param bytes 参数以数据流形式发送
	 * @return 响应返回内容
	 */
	public static String requestPostByBytes(String url, String encoding, int timeout, byte[] bytes) throws Exception {
		return requestHttp(url, encoding, HttpHelper.METHOD_POST, timeout, null, new ByteArrayInputStream(bytes));
	}
}
