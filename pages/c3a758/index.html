Java JSpinner

# JSpinner

微调控制器，JSpinner类的对象是一个单行输入字段，允许用户从有序序列中选择一个数字或对象值。

# 常用构造函数：

Constructor Description
JSpinner() 它用于构造一个带有整数SpinnerNumberModel的微调器，初始值为0，没有最小或最大限制。
JSpinner(SpinnerModel model) 它用于为给定模型构造旋转器。

# 常用方法：

Method Description
void addChangeListener(ChangeListener listener) 它用于将侦听器添加到列表中，每当模型发生更改时都会通知该列表。
Object getValue() 它用于返回模型的当前值。

# 示例代码：

package code.swing.learn;

import javax.swing.*;

/**
* @author :ghgeng
* @date 2023/7/17 18:05
*/
public class JSpinnerTest {


   public static void main(String[] args) {
       JFrame j = new JFrame("JSpinnerTest");

       SpinnerModel spinnerModel = new SpinnerNumberModel(
               5, // 初始化大小
               0,// 最小值
               10,// 最大值
               1// 步数
       );

       JSpinner jSpinner = new JSpinner(spinnerModel);
       jSpinner.setBounds(30, 30, 100, 30);

       j.add(jSpinner);
       // 显示窗口
       j.setSize(400, 400);
       j.setLayout(null);
       j.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
       j.setLocationRelativeTo(null);
       j.setVisible(true);
   }
}

# 示例代码2：ChangeListener

package code.swing.learn;

import javax.swing.*;

/**
 * @author :ghgeng
 * @date 2023/7/17 18:05
 */
public class JSpinnerTest {


    public static void main(String[] args) {
        JFrame j = new JFrame("JSpinnerTest");
        JLabel jLabel = new JLabel();
        jLabel.setBounds(100, 100, 50, 30);
        j.add(jLabel);

        SpinnerModel spinnerModel = new SpinnerNumberModel(
                5, // 初始化大小
                0,// 最小值
                10,// 最大值
                1// 步数
        );

        JSpinner jSpinner = new JSpinner(spinnerModel);
        jSpinner.setBounds(30, 30, 100, 30);
        jSpinner.addChangeListener(e ->{
            Object value = jSpinner.getValue();
            jLabel.setText("value: " + value);

        });


        j.add(jSpinner);
        // 显示窗口
        j.setSize(400, 400);
        j.setLayout(null);
        j.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        j.setLocationRelativeTo(null);
        j.setVisible(true);
    }
}
