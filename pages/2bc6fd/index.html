Easy Excel

# 简单使用

package code.test;

import com.alibaba.excel.EasyExcel;
import com.alibaba.excel.support.ExcelTypeEnum;
import com.alibaba.excel.write.style.column.SimpleColumnWidthStyleStrategy;

import java.io.FileOutputStream;
import java.io.OutputStream;
import java.util.ArrayList;
import java.util.Arrays;
import java.util.List;

/**
 * @author :ghgeng
 * @date 2023/11/17 17:55
 */
public class EasyExcle {

    public static void main(String[] args) throws Exception {
        // 文件输出位置
        OutputStream out = new FileOutputStream("D://测试.xlsx");
        EasyExcel.write(out).needHead(true)
                .head(head()) //表头
                .excelType(ExcelTypeEnum.XLSX)
                 // 设置宽度
                .registerWriteHandler(new SimpleColumnWidthStyleStrategy(20))
                .sheet(0, "城市介绍")
                 // 内容
                .doWrite(contentData());
    }

    private static List<List<String>> head() {
        List<List<String>> headTitles = new ArrayList<>();
        headTitles.add(Arrays.asList("城市"));
        headTitles.add(Arrays.asList("人口数目"));
        headTitles.add(Arrays.asList("出行方式"));
        headTitles.add(Arrays.asList("著名景点"));
        return headTitles;
    }

    private static List<List<String>> contentData() {
        List<List<String>> data = new ArrayList<>();
        data.add(Arrays.asList("深圳", "200", "地铁", "深圳湾"));
        data.add(Arrays.asList("广州", "100", "地铁", "珠江"));
        data.add(Arrays.asList("北京", "300", "地铁", "故宫"));
        data.add(Arrays.asList("杭州", "400", "地铁", "西湖"));
        return data;

    }
}
