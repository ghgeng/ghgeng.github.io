JavaFX 灯光效果

提示

JavaFX Lighting
此效果用于从光源点亮节点。光源有很多种，即点光源、远光源和点光源。类javafx.scene.effect.Lighting表示灯光效果。我们需要实例化这个类，以便在节点上生成适当的效果。

# 属性

方法 描述
setBumpInput(Effect value) 它是一个Effect对象类型属性。它表示效果的凹凸贴图输入。
setContentInput(Effect value) 它是一个Effect对象类型属性。它代表了Effect的内容输入。
setDiffuseConstant(Double value) 它是Double类型属性。它表示扩散常数。
setLight(Light value) 它是一个Light对象类型属性。它代表了效果的光源。
setSpecularConstant(double value) 它是一个双类型属性。它表示镜面常数。
setSpecularExponent(double value) 它是一个双类型属性。它代表镜面指数。
setSurfaceScale(double value) 它是双类型属性。它表示光的表面尺度。

# 示例1

package code.javafx.JavaFXEffects;

import javafx.application.Application;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.effect.BlurType;
import javafx.scene.effect.InnerShadow;
import javafx.scene.effect.Lighting;
import javafx.scene.image.Image;
import javafx.scene.image.ImageView;
import javafx.scene.paint.Color;
import javafx.scene.shape.Circle;
import javafx.scene.text.Font;
import javafx.scene.text.FontWeight;
import javafx.scene.text.Text;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2024/1/29 15:07
 */
public class JavaLightingExample extends Application {

    @Override
    public void start(Stage primaryStage) throws Exception {
        primaryStage.setTitle("JavaLightingExample");

        Text text = new Text();
        text.setFont(Font.font(null, FontWeight.BOLD, 35));
        text.setX(60);
        text.setY(40);
        text.setText("Hello World");
        text.setFill(Color.GREEN);

        Circle circle = new Circle();
        //Setting the center of the circle
        circle.setCenterX(300.0f);
        circle.setCenterY(160.0f);
        //Setting the radius of the circle
        circle.setRadius(100.0f);
        //setting the fill color of the circle
        circle.setFill(Color.GRAY);
        //Instantiating the Lighting class
        Lighting lighting = new Lighting();
        //Applying lighting effect to the text
        text.setEffect(lighting);
        //Applying lighting effect to the circle
        circle.setEffect(lighting);

        Group root = new Group(text,circle);
        Scene scene = new Scene(root, 600, 350);
        primaryStage.setScene(scene);
        primaryStage.show();
    }

    public static void main(String[] args) {
        launch(args);
    }

}
