# JLabel 组件

JLabel:标签控件，用于显示文本
JLabel类的对象是一个用于在容器中放置文本的组件。它用于显示单行只读文本。文本可以由应用程序更改，但用户不能直接编辑它。它继承了JComponent类。

# 常用构造函数:

Constructor Description
JLabel() 创建一个JLabel实例，没有图像，标题为空字符串。
JLabel(String s) 创建具有指定文本的JLabel实例。
JLabel(Icon i) 使用指定的图像创建JLabel实例。
JLabel(String s, Icon i, int horizontalAlignment) 创建具有指定文本、图像和水平对齐方式的JLabel实例。

# 常用方法

Methods Description
String getText() T返回标签显示的文本字符串。
void setText(String text) 它定义了该组件将显示的单行文本
void setHorizontalAlignment(int alignment) 它设置标签内容沿X轴的对齐方式
Icon getIcon() 它返回标签显示的图形图像
int getHorizontalAlignment() 它返回标签内容沿X轴的对齐方式

package code.swing.learn;

import javax.swing.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

/**
 * @author :ghgeng
 * @date 2023/6/3 9:40
 */
public class MySwing extends JFrame {

    private static JPanel root = new JPanel();

    public static void main(String[] args) {
        MySwing mySwing = new MySwing();
        mySwing.initMyFrame();
        mySwing._03();
    }

    public void initMyFrame() {
        this.setTitle("第一个窗口");
        // 当关闭窗口时 ，默认关闭程序
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        // 设置窗口(组件)大小
        this.setSize(400, 600);
        // 居中展示
        this.setLocationRelativeTo(null);
        // 设置面板，所有的组件都可以 .add() 放上去
        // 就好比有一个画板，画画之前放一张纸上去，大小可以自己设置
        this.setContentPane(root);


    }
    /**
     * 容器与控件
     * JLabel:标签控件，用于显示文本
     */
    public void _03() {
        // 创建一个 lable组件，放到面板中
        JLabel jLabel = new JLabel("Hello World");
        root.add(jLabel);
        // 创建一个按钮组件放到面板中
        JButton jButton = new JButton("测试");
        root.add(jButton);
        // 创建lable组件直接放到面板中
        root.add(new JLabel("Hello1 World1"));
        // 显示窗口
        this.setVisible(true);
    }
}

# 示例代码2：

package code.swing.learn;

import lombok.SneakyThrows;

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.net.InetAddress;

/**
 * @author :ghgeng
 * @date 2023/7/3 22:13
 */
public class JLabelTest extends JFrame implements ActionListener {

    JTextField jTextField;
    JLabel jlable;
    JButton jButton;

    JLabelTest() {
        jTextField = new JTextField("www.baidu.com");
        jTextField.setBounds(50, 10, 150, 30);
        this.add(jTextField);

        jlable = new JLabel();
        jlable.setBounds(50, 150, 400, 30);
        this.add(jlable);


        jButton = new JButton("Find IP");
        jButton.setBounds(50, 70, 150, 30);
        jButton.addActionListener(this);
        this.add(jButton);

        this.setLayout(null);
        this.setSize(400, 300);
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        this.setLocationRelativeTo(null);
        this.setVisible(true);
    }


    public static void main(String[] args) {
        new JLabelTest();
    }

    @SneakyThrows
    @Override
    public void actionPerformed(ActionEvent e) {
        String text = jTextField.getText();
        String hostName = InetAddress.getByName(text).getHostAddress();
        jlable.setText("Host :" + text + "IP : " + hostName);

    }
}
