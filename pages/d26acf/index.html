javaFX Radio Button

# Radio Button

提示

单选按钮用于向用户提供各种选项。用户只能在所有选项中选择一个。单选按钮被选中或取消选中。它可用于测验中的多项选择题，学生只需选择一个选项。

package code.javafx.JavaFXUI;

import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.control.RadioButton;
import javafx.scene.control.ToggleGroup;
import javafx.scene.layout.StackPane;
import javafx.scene.layout.VBox;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2024/7/5 19:36
 */
public class RadioButtonTest extends Application {
    @Override
    public void start(Stage primaryStage) throws Exception {

        VBox root = new VBox();


        ToggleGroup group = new ToggleGroup();
        RadioButton button1 = new RadioButton("option 1");
        RadioButton button2 = new RadioButton("option 2");
        RadioButton button3 = new RadioButton("option 3");
        RadioButton button4 = new RadioButton("option 4");
        button1.setToggleGroup(group);
        button2.setToggleGroup(group);
        button3.setToggleGroup(group);
        button4.setToggleGroup(group);

        Scene scene = new Scene(root, 300, 300);

        root.getChildren().addAll(button1, button2, button3, button4);
        primaryStage.setScene(scene);
        primaryStage.setTitle("Radio Class Example");
        primaryStage.show();

    }

    public static void main(String[] args) {
        launch(args);
    }
}
