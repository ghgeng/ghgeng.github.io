Java JColorChooser

# JColorChooser

JColorChooser类用于创建一个颜色选择器对话框，以便用户可以选择任何颜色。它继承了JComponent类。

# 常用构造函数：

Constructor Description
JColorChooser() 它最初用于创建一个白色的颜色选择器面板。
JColorChooser(color initialcolor) 它用于创建一个带有指定颜色的颜色选择面板。

# 常用方法:

Method Description
void addChooserPanel(AbstractColorChooserPanel panel) 它用于向颜色选择器添加颜色选择器面板。
static Color showDialog(Component c, String title, Color initialColor) 它用于显示颜色选择器对话框。

# 示例代码1：

package code.swing.learn;

import com.sun.webkit.ColorChooser;

import javax.swing.*;
import java.awt.*;

/**
 * @author :ghgeng
 * @date 2023/7/17 16:19
 */
public class JColorChooserTest extends JFrame {

    JColorChooserTest() {
        JButton jButton = new JButton("color");
        jButton.addActionListener(e -> {
            Color select_a_color = JColorChooser.showDialog(this, "select a color", Color.red);
            this.setBackground(select_a_color);
        });

        // 显示窗口
        this.add(jButton);
        this.setSize(400, 400);
        this.setLayout(new FlowLayout());
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        this.setLocationRelativeTo(null);
        this.setVisible(true);
    }

    public static void main(String[] args) {
        new JColorChooserTest();
    }
}

# 示例代码2：

package code.swing.learn;

import com.sun.webkit.ColorChooser;

import javax.swing.*;
import java.awt.*;

/**
 * @author :ghgeng
 * @date 2023/7/17 16:19
 */
public class JColorChooserTest extends JFrame {

    JColorChooserTest() {
        // 创建文本域
        JTextArea jTextArea = new JTextArea("test color");
        jTextArea.setBounds(30, 30, 200, 200);
        this.add(jTextArea);

        JButton jButton = new JButton("color");
        jButton.setBounds(100, 300, 100, 30);
        jButton.addActionListener(e -> {
            Color select_a_color = JColorChooser.showDialog(this, "select a color", Color.red);
            jTextArea.setBackground(select_a_color);
        });

        // 显示窗口
        this.add(jButton);
        this.setSize(400, 400);
        this.setLayout(null);
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        this.setLocationRelativeTo(null);
        this.setVisible(true);
    }

    public static void main(String[] args) {
        new JColorChooserTest();
    }
}
