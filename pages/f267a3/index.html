JavaFX Polygons

提示

多边形可以定义为至少有三条直边构成一个环的平面图形。对于多边形，我们主要考虑它的边长和内角。三角形、正方形、五边形、六边形等都是多边形。
在JavaFX中，Polygon可以通过实例化JavaFX.scene.shape.Polygon类来创建。我们需要向类构造函数传递一个Double数组，表示多边形所有点的X-Y坐标。语法如下所示。

Polygon poly = new Polygon(DoubleArray);
1

我们也可以通过调用getPoints()方法返回的引用匿名调用addAll()方法来创建多边形，getPoints()方法是polygon类的实例方法。然而，我们需要将双数组传递给这个方法，它表示多边形的X-Y坐标。语法如下所示。

Pollygon polygon_object = new Pollygon();  
Pollygon_Object.getPoints().addAll(Double_Array);
1
2

# 示例

创建多边形

package code.javafx;

import javafx.application.Application;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.paint.Color;
import javafx.scene.shape.Circle;
import javafx.scene.shape.Polygon;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2023/9/20 17:39
 */
public class JavaFXPolygons extends Application {

    @Override
    public void start(Stage primaryStage) throws Exception {
        primaryStage.setTitle("JavaFXPolygons");
        Group group = new Group();
        // 第一种创建方法
/*
        double d[] = {
                0.0,0.0,
                100.0,200.0,
                200.0,100.0
                    };
        Polygon polygon = new Polygon(d);
*/

        // 第二种多边形方法
        Double d[] = new Double[]{
                0.0,0.0,
                100.0,200.0,
                200.0,100.0
        };
        Polygon polygon = new Polygon();
        polygon.getPoints().addAll(d);

        group.getChildren().addAll(polygon);
        Scene scene = new Scene(group, 300, 300, Color.ANTIQUEWHITE);
        primaryStage.setScene(scene);
        primaryStage.show();
    }


    public static void main(String[] args) {
        launch(args);
    }
}
