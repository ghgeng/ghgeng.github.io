# 网络编程有哪些其他类

  • 1、Socket 类：
    • 用于客户端的网络编程，通过它可以连接到服务器，并发送和接收数据。
    • 提供了 connect() 方法用于连接到服务器，以及 getOutputStream() 和 getInputStream() 方法用于获取输出和输入流。
  • 1、ServerSocket 类：
    • 用于服务器端的网络编程，它监听一个指定的端口，等待客户端的连接。
    • 提供了 accept() 方法，该方法会阻塞直到一个连接建立，然后返回一个 Socket 对象，用于与客户端通信。
  • 2、DatagramSocket 类：
    • 用于无连接的数据报套接字编程，通常用于 UDP 协议。
    • 提供了 send() 方法发送数据报，以及 receive() 方法接收数据报。
  • 2、DatagramPacket 类：
    • 表示一个数据报包，它包含了要发送或接收的数据，以及数据的长度、目标地址和端口等信息。
    • 通常与 DatagramSocket 类一起使用。
  • 3、URL 和 URLConnection 类：
    • URL 类表示一个统一资源定位符，它指向网络上的一个资源。
    • URLConnection 类表示一个到 URL 指向的资源的连接，通过它可以读取或写入资源。
  • 3、HttpURLConnection 类：
    • 是 URLConnection 的一个子类，专门用于 HTTP 协议的网络通信。
    • 提供了更高级别的 HTTP 请求和响应处理功能。
  • 4、ServerSocketChannel 和 SocketChannel 类（Java NIO）：
    • 这些类属于 Java 的 New I/O（NIO）库，提供了基于通道（Channel）的非阻塞网络通信。
    • ServerSocketChannel 用于服务器端，SocketChannel 用于客户端。
  • 4、Selector 类（Java NIO）：
    • 用于同时监控多个通道（Channel）的事件，如连接、读取、写入等。
    • 使得单个线程可以管理多个网络连接，提高了网络编程的效率和可扩展性。
  • 5、RMI（Remote Method Invocation）相关类：
    • 如 java.rmi 包中的 Remote 接口、Naming 类等，用于实现分布式对象调用。
    • 允许不同计算机上的 Java 应用程序通过网络相互调用对象中的方法。

# 1、简单代码示例

接收方代码

package code.test;

import cn.hutool.core.date.DateUtil;
import lombok.AllArgsConstructor;
import lombok.Data;

import java.net.DatagramPacket;
import java.net.DatagramSocket;
import java.net.InetAddress;
import java.net.SocketException;

/**
 * @author :ghgeng
 * @date 2024/10/14 15:59
 */
@Data
@AllArgsConstructor
public class Receive {
    // 端口号，根据业务修改
    private int port;

    public static void main(String[] args) throws Exception {
        // 远程地址ip和端口号，还有发送方名称
        // 接受方作为服务端的话不需要发送方host，本机ip就是host
        new Receive(9999).startReceive();
    }

    public void startReceive() throws Exception {
        // 新建报文连接，等于服务端
        DatagramSocket datagramSocket = new DatagramSocket(port);
        // 接受多少个字节
        byte[] bytes = new byte[1024];

        DatagramPacket datagramPacket = new DatagramPacket(bytes, 0, bytes.length);
        // 接受数据，此方法阻塞线程
        datagramSocket.receive(datagramPacket);

        byte[] resData = datagramPacket.getData();

        InetAddress address = datagramPacket.getAddress();
        System.out.println(DateUtil.now());
        System.out.printf(" 接受内容来自%s内容：%n %s",
                address.getHostAddress(),
                new String(resData, 0, resData.length));
    }
}

发送方代码

package org.example.demo.test;

import lombok.AllArgsConstructor;
import lombok.Data;

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.net.DatagramPacket;
import java.net.DatagramSocket;
import java.net.InetAddress;

@Data
@AllArgsConstructor
public class Send {
	// 远程ip
	private String hostIp;
	// 远程端口号
	private int prot;
	// 发送方名称
	private String msgName;


	public static void main(String[] args) throws Exception {
		new Send("192.168.61.59", 9999, "学生").send();
	}

	void send() throws Exception {
		// 获取学生输入
		BufferedReader bufferedReader = new BufferedReader(new InputStreamReader(System.in));
		String s = bufferedReader.readLine();
		// 转成字节数组
		byte[] bytes = (this.msgName + ":" + s).getBytes();

		// 建立socket链接;
		DatagramSocket datagramSocket = new DatagramSocket();

		// 通过数据报设置的ip 发送给制定的客户端
		InetAddress remoteHost = InetAddress.getByName(this.hostIp);
		DatagramPacket datagramPacket = new DatagramPacket(bytes, 0, bytes.length, remoteHost, this.prot);

		datagramSocket.send(datagramPacket);
	}
}
