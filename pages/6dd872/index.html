# JCheckBox 复选框组件

JCheckBox :复选框组件 JCheckBox类用于创建一个复选框。它用于on(true)或off(false)选项。点击复选框将其状态从“on”变为“off”或从“off”变为“on”。它继承了JToggleButton类。

# 常用构造函数:

Constructor Description
JJCheckBox() 创建一个初始未选中的复选框按钮，没有文本，没有图标。
JChechBox(String s) 创建带有文本的初始未选中复选框。
JCheckBox(String text, boolean selected) 创建带有文本的复选框，并指定最初是否选中该复选框。
JCheckBox(Action a) 创建一个复选框，其中的属性取自提供的Action。

# 常用方法

Methods Description
AccessibleContext getAccessibleContext() 它用于获取与该JCheckBox关联的AccessibleContext。
protected String paramString() 它返回这个JCheckBox的字符串表示形式

package code.swing.learn;
import javax.swing.*;
import java.awt.event.ItemEvent;
import java.awt.event.ItemListener;

/**
 * @author :ghgeng
 * @date 2023/7/5 21:53
 */
public class JcheckBoxTest {
    public static void main(String[] args) {
        JFrame j = new JFrame("JCheckBox");

        // 简单的容器
        JPanel jPanel = new JPanel();
        JCheckBox jCheckBox = new JCheckBox("是否同意", true);
        // 是否能点击 ,默认 true
        jPanel.add(jCheckBox);

        // 下一步按钮操作
        JButton jButton = new JButton("下一步");
        // 默认不能点击，等复选框点击才放开按钮
        jButton.setEnabled(false);
        jPanel.add(jButton);

        // 添加事件
        jCheckBox.addActionListener((e) -> {
            // 是否被选中了
            if (jCheckBox.isSelected()) {
                jButton.setEnabled(true);
            } else {
                jButton.setEnabled(false);
            }
        });

        // 显示窗口
        j.setSize(400, 400);
        j.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        j.setLocationRelativeTo(null);
        j.setContentPane(jPanel);
        j.setVisible(true);


    }
}

# 示例代码3：

package code.swing.learn;

import javax.swing.*;
import java.awt.event.ItemEvent;
import java.awt.event.ItemListener;

/**
 * @author :ghgeng
 * @date 2023/7/5 21:53
 */
public class JcheckBoxTest {

    public static void main(String[] args) {
        JFrame j = new JFrame("JCheckBox");

        JLabel jLabel = new JLabel();
        jLabel.setSize(400, 400);
        jLabel.setBounds(50, 50, 200, 50);
        jLabel.setLayout(null);
        j.add(jLabel);

        // 新建两个复选框
        JCheckBox jCheckBox1 = new JCheckBox("C++");
        jCheckBox1.setBounds(100, 100, 100, 50);
        j.add(jCheckBox1);

        JCheckBox jCheckBox2 = new JCheckBox("Java");
        jCheckBox2.setBounds(100, 150, 100, 50);
        j.add(jCheckBox2);

        // 给复选框加事件处理
        jCheckBox1.addItemListener(e -> {
            // 是否被选中
            int stateChange = e.getStateChange();
            if (stateChange == 1) {
                jLabel.setText("C++ Checkbox:checked" );
            } else {
                jLabel.setText("C++ Checkbox:unchecked" );
            }

        });

        // 给复选框加事件处理
        jCheckBox2.addItemListener(e -> {
            // 是否被选中  1 被选中  2 取消选中
            int stateChange = e.getStateChange();
            if (stateChange == 1) {
                jLabel.setText("Java Checkbox:checked" );
            } else {
                jLabel.setText("Java  Checkbox:unchecked" );
            }

        });


        j.setSize(400, 400);
        j.setLayout(null);
        j.setLocationRelativeTo(null);
        j.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        j.setVisible(true);


    }
}
