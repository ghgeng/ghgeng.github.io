JavaFX ColorInput

提示

JavaFX ColorInput
ColorInput产生类似于彩色矩形的输出。它不显示节点，而是显示矩形框。它主要作为输入传递给其他效果。类javafx.scene.effect.ColorInput表示ColorInput效果。该类的对象作为其他效果的输入传递。

# 属性

方法 描述
setHeight(double value) 它是双型的。它表示要填充的区域的高度。
setPaint(Paint value) 它表示要填充该区域的油漆。
setWidth(double value) 它是双型的。它表示要填充的区域的宽度。
setX(double value) 它表示区域左上角的X坐标。
setY(double value) 它表示区域左上角的Y坐标。

# 示例1

package code.javafx.JavaFXEffects;

import javafx.application.Application;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.effect.ColorInput;
import javafx.scene.paint.Color;
import javafx.scene.shape.Rectangle;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2023/9/20 17:39
 */
public class JavaFXColorInput extends Application {

    @Override
    public void start(Stage primaryStage) throws Exception {
        primaryStage.setTitle("JavaFXColorInput");
        ColorInput colorInput = new ColorInput();
        colorInput.setPaint(Color.RED);
        colorInput.setHeight(100);
        colorInput.setWidth(100);
        colorInput.setX(100);
        colorInput.setY(100);

        // 创建矩形框
        Rectangle rectangle = new Rectangle();
        // 设置特效会把之前的覆盖
/*        rectangle.setX(20);
        rectangle.setY(20);
        rectangle.setWidth(1000);
        rectangle.setHeight(100);*/
        rectangle.setEffect(colorInput);

        Group root = new Group();
        root.getChildren().addAll(rectangle);
        Scene scene = new Scene(root, 700, 400);
        primaryStage.setScene(scene);
        primaryStage.show();
    }

    public static void main(String[] args) {
        launch(args);
    }
}
