Java JSeparator

# JSeparator

JSeparator类的对象用于提供实现分隔线的通用组件。它用于绘制一条线来分隔布局中的小部件。它继承了JComponent类。

# 常用构造函数：

Constructor Description
JSeparator() 创建一个新的水平分隔符。
JSeparator(int orientation) 创建具有指定水平或垂直方向的新分隔符。

# 常用方法：

Method Description
void setOrientation(int orientation) 它用于设置分离器的方向。
int getOrientation() 它用于返回分隔符的方向。

# 示例代码1：

package code.swing.learn;

import javax.swing.*;

/**
 * @author :ghgeng
 * @date 2023/7/17 11:27
 */
public class JSeparatorTest {
    public static void main(String[] args) {
        JFrame j = new JFrame("JSeparatorTest");
        // 创建导航菜单
        JMenuBar jMenuBar = new JMenuBar();
        // 创建菜单
        JMenu jMenu = new JMenu("File");
        // 创建子菜单
        JMenuItem jMenuItem1 = new JMenuItem("Copy");
        JMenuItem jMenuItem2 = new JMenuItem("Cut");
        JMenuItem jMenuItem3 = new JMenuItem("Paste");
            // 添加子菜单
            jMenu.add(jMenuItem1);
            jMenu.addSeparator();
            jMenu.add(jMenuItem2);
            jMenu.add(jMenuItem3);

        jMenuBar.add(jMenu);
        j.setJMenuBar(jMenuBar);

        // 显示窗口
        j.setSize(400, 400);
        j.setLayout(null);
        j.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        j.setLocationRelativeTo(null);
        j.setVisible(true);
    }
}

# 示例代码2：

package code.swing.learn;

import javax.swing.*;
import java.awt.*;

/**
 * @author :ghgeng
 * @date 2023/7/17 11:27
 */
public class JSeparatorTest {
    public static void main(String[] args) {
        JFrame j = new JFrame("JSeparatorTest");

        JLabel jLabel1 = new JLabel("Above Separator");
        j.add(jLabel1);
        // 添加分割
        JSeparator jSeparator = new JSeparator();
        j.add(jSeparator);
        JLabel jLabel2 = new JLabel("Above Separator");
        j.add(jLabel2);

        // 显示窗口
        j.setSize(400, 400);
        j.setLayout(new GridLayout(0,1));
        j.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        j.setLocationRelativeTo(null);
        j.setVisible(true);
    }
}
