JavaFX Pane

# Pane

提示

在JavaFX中，Pane 是一个基类，用于所有布局容器，但它本身不提供任何特定的布局算法。它仅仅是一个容器，允许您在其中放置子节点，但不会自动调整这些子节点的位置或大小。然而，您可以通过编程方式直接设置子节点的布局属性（如位置和大小）。

# 示例代码

import javafx.application.Application;  
import javafx.scene.Scene;  
import javafx.scene.control.Button;  
import javafx.scene.layout.Pane;  
import javafx.stage.Stage;  
  
public class PaneExample extends Application {  
  
    @Override  
    public void start(Stage primaryStage) {  
        // 创建一个Pane  
        Pane pane = new Pane();  
        pane.setPrefSize(300, 200); // 设置Pane的首选大小  
  
        // 创建一些按钮  
        Button button1 = new Button("Button 1");  
        Button button2 = new Button("Button 2");  
  
        // 设置按钮的位置  
        // 注意：这里的坐标是相对于Pane的左上角的  
        button1.setLayoutX(50); // 设置button1的X坐标  
        button1.setLayoutY(30); // 设置button1的Y坐标  
  
        button2.setLayoutX(150); // 设置button2的X坐标  
        button2.setLayoutY(70); // 设置button2的Y坐标  
  
        // 将按钮添加到Pane中  
        pane.getChildren().add(button1);  
        pane.getChildren().add(button2);  
  
        // 设置场景和舞台  
        Scene scene = new Scene(pane);  
  
        primaryStage.setTitle("Pane Example"); // 舞台标题  
        primaryStage.setScene(scene); // 将场景设置给舞台  
        primaryStage.show(); // 显示舞台  
    }  
  
    public static void main(String[] args) {  
        launch(args); // 启动JavaFX应用程序  
    }  
}
