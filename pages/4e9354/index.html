JavaFX GridPane

# GridPane

提示

GridPane 是JavaFX中一个非常灵活的布局容器，它允许您以网格形式组织子节点，通过行和列来定位每个节点。

  • SceneBuilder布局

# 示例代码

package code.javafx.learn.JavaFX布局;

import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.control.Button;
import javafx.scene.control.Label;
import javafx.scene.layout.GridPane;
import javafx.stage.Stage;

public class GridPaneExample extends Application {

    @Override
    public void start(Stage primaryStage) {
        // 创建一个GridPane
        GridPane gridPane = new GridPane();
        gridPane.setHgap(10); // 设置水平间距
        gridPane.setVgap(10); // 设置垂直间距
        gridPane.setPadding(new javafx.geometry.Insets(10)); // 设置内边距

        // 创建一些节点并添加到GridPane中
        // 添加标签
        Label label1 = new Label("First Name:");
        Label label2 = new Label("Last Name:");
        gridPane.add(label1, 0, 0); // 添加到第0行第0列
        gridPane.add(label2, 0, 1); // 添加到第1行第0列

        // 添加按钮
        Button button1 = new Button("John");
        Button button2 = new Button("Doe");
        gridPane.add(button1, 1, 0); // 添加到第0行第1列
        gridPane.add(button2, 1, 1); // 添加到第1行第1列

        // 设置场景和舞台
        Scene scene = new Scene(gridPane, 600, 400);

        primaryStage.setTitle("GridPane Example"); // 舞台标题
        primaryStage.setScene(scene); // 将场景设置给舞台
        primaryStage.show(); // 显示舞台
    }

    public static void main(String[] args) {
        launch(args); // 启动JavaFX应用程序
    }
}
