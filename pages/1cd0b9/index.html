Java JPasswordField

# JPasswordField

JPasswordField类的对象是一个专门用于密码输入的文本组件。它允许编辑单行文本。它继承了JTextField类。

# 常用构造方法

Constructor Description
JPasswordField() 构造一个新的JPasswordField，包含默认文档、空起始文本字符串和0列宽度。
JPasswordField(int columns) 构造具有指定列数的新的空JPasswordField。
JPasswordField(String text) 构造一个用指定文本初始化的新JPasswordField。
JPasswordField(String text, int columns) 构造一个用指定的文本和列初始化的新JPasswordField。

# 示例代码1:

package code.swing.learn;

import javax.swing.*;

/**
 * @author :ghgeng
 * @date 2023/7/6 22:21
 */
public class JPasswordFieldText {

    public static void main(String[] args) {
        JFrame jFrame = new JFrame("JPasswordField");

        JLabel jLabel = new JLabel("password:");
        jLabel.setBounds(20, 20, 80, 30);

        JPasswordField jPasswordField = new JPasswordField();
        jPasswordField.setBounds(100, 20, 150, 25);
        jPasswordField.setText("123");

        jFrame.add(jLabel);
        jFrame.add(jPasswordField);

        jFrame.setLayout(null);
        jFrame.setSize(400, 400);
        jFrame.setLocationRelativeTo(null);
        jFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        jFrame.setVisible(true);
    }
}

# 示例代码2：

package code.swing.learn;

import javax.swing.*;

/**
 * @author :ghgeng
 * @date 2023/7/6 22:21
 */
public class JPasswordFieldText {

    public static void main(String[] args) {
        JFrame jFrame = new JFrame("JPasswordField");

        JLabel jLabel = new JLabel("username:");
        jLabel.setBounds(20, 20, 100, 30);

        JLabel jLabe2 = new JLabel("password:");
        jLabe2.setBounds(20, 70, 100, 30);

        JLabel jLabel3 = new JLabel();
        jLabel3.setBounds(20, 200, 200, 30);

        JTextField jTextField = new JTextField();
        jTextField.setBounds(100, 20, 150, 30);

        JPasswordField jPasswordField = new JPasswordField();
        jPasswordField.setBounds(100, 70, 150, 30);

        JButton jButton = new JButton("login");
        jButton.setBounds(100, 120, 100, 30);
        jButton.addActionListener(e -> {
            String username = jTextField.getText();
            String password = new String(jPasswordField.getPassword());
            jLabel3.setText("username: " + username + "\n" + "password: " + password);

        });

        jFrame.add(jLabel);
        jFrame.add(jLabe2);
        jFrame.add(jLabel3);
        jFrame.add(jTextField);
        jFrame.add(jPasswordField);
        jFrame.add(jButton);

        jFrame.setLayout(null);
        jFrame.setSize(400, 400);
        jFrame.setLocationRelativeTo(null);
        jFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        jFrame.setVisible(true);
    }
}
