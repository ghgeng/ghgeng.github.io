JavaFX Gradient Color

# JavaFX渐变颜色

  • 在计算机图形学中，渐变颜色(有时称为颜色渐变)用于指定填充特定区域的位置相关颜色。渐变颜色的值随位置的变化而变化。渐变颜色通过随位置连续变化颜色值来产生区域上平滑的颜色过渡。
  • JavaFX使我们能够实现两种类型的渐变颜色过渡:
    • 线性渐变
    • 径向梯度

# 线性渐变

要将线性梯度模式应用到形状上，我们需要实例化LinearGradient类。这个类包含下面表中描述的几个实例方法。

# 示例1

package code.javafx;

import javafx.application.Application;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.layout.VBox;
import javafx.scene.paint.Color;
import javafx.scene.paint.CycleMethod;
import javafx.scene.paint.LinearGradient;
import javafx.scene.paint.Stop;
import javafx.scene.shape.Rectangle;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2023/9/20 17:39
 */
public class JavaFXGradientColor extends Application {

    @Override
    public void start(Stage primaryStage) throws Exception {
        primaryStage.setTitle("JavaFXGradientColor");

        VBox root = new VBox();
        Stop[] stops = new Stop[]{new Stop(0, Color.GREEN), new Stop(1, Color.ANTIQUEWHITE)};
        // (startX,startY):表示渐变颜色起始点的x、y坐标。
        // (endX,endY):表示渐变颜色终点的x、y坐标。
        // Proportional比例:这是一个布尔类型属性。如果这是真的，那么渐变颜色的起始点和结束点将成比例。
        // CycleMethod:这定义了应用于梯度的循环方法。
        // stops:这定义了沿梯度的颜色分布。
        LinearGradient linear = new LinearGradient(0, 0, 1, 0, true, CycleMethod.NO_CYCLE, stops);

        Rectangle rect = new Rectangle(0, 0, 100, 100);
        rect.setFill(linear);

        root.getChildren().addAll(rect);
        Scene scene = new Scene(root, 300, 250, Color.ANTIQUEWHITE);

        primaryStage.setScene(scene);
        primaryStage.show();
    }


    public static void main(String[] args) {
        launch(args);
    }
}

# 径向渐变

要对形状应用径向梯度，我们需要实例化javafx.scene.paint.RadialGradient类。这个类包含下表中描述的各种实例方法。

package code.javafx;

import javafx.application.Application;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.layout.VBox;
import javafx.scene.paint.*;
import javafx.scene.shape.Circle;
import javafx.scene.shape.Rectangle;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2023/9/20 17:39
 */
public class JavaFXGradientColor extends Application {

    public static void main(String[] args) {
        launch(args);
    }

    @Override
    public void start(Stage primaryStage) throws Exception {
        primaryStage.setTitle("Radial Gradient Example");
        Group root = new Group();

        Scene scene = new Scene(root, 400, 300, Color.WHITE);

        primaryStage.setScene(scene);
        addRectangle(scene);
        primaryStage.show();
    }


    private void addRectangle(final Scene scene) {
        Circle C = new Circle(200, 150, 100);
        RadialGradient gradient1 = new RadialGradient(0,
                0.1,
                100,
                100,
                200,
                false,
                CycleMethod.NO_CYCLE,
                new Stop(0, Color.YELLOW),
                new Stop(1, Color.RED));

        C.setFill(gradient1);
        Group root = (Group) scene.getRoot();
        root.getChildren().add(C);
    }
}
