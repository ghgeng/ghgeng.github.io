package utils.jdbc;
import java.sql.Connection;
import java.sql.DriverManager;

public class JBDCUtil {
	// 数据库连接
	private final static String url = "jdbc:mysql://172.16.102.70:3306/test";
	// 账号
	private static final String password = "123456";
	// 密码
	private static final String user = "root";

	private volatile static Connection conn;


	/**
	 * 正常连接
	 */
//	public static Connection getConnection() {
//		try {
//			// 1.加载驱动
//			Class.forName("com.mysql.jdbc.Driver");
//			// 2.获取连接
//			return DriverManager.getConnection(url, user, password);
//		} catch (Exception e) {
//			e.printStackTrace();
//		}
//		return null;
//	}


	/**
	 * 双重校验
	 * @return
	 */
	public static Connection getConnection() {
		try {
			if (conn == null) {
				synchronized (JBDCUtil.class) {
					if (conn == null) {
						Class.forName("com.mysql.jdbc.Driver");
						conn = DriverManager.getConnection(url, user, password);
					}
				}
			}
			return conn;
		} catch (Exception e) {
			e.printStackTrace();
		}
		return conn;
	}
}

import java.sql.CallableStatement;
import java.sql.Connection;
import java.sql.PreparedStatement;
import java.sql.ResultSet;

public class Test {
	public static void main(String[] args) throws Exception {
		PreparedStatement statement = null;
		ResultSet resultSet = null;
		Connection connection = null;
		try {
			connection = JBDCUtil.getConnection();
            String sql = "select status,start_time,end_time from execution_flows where exec_id = '4841';";
            //预编译SQl
            statement = connection.prepareStatement(sql);
            resultSet = statement.executeQuery();
            while (resultSet.next()) {
                String status = resultSet.getString("status");
                String start_time = resultSet.getString("start_time");
                String end_time = resultSet.getString("end_time");
                System.out.println(status+ "  " +start_time + "  " +end_time);
            }
			
		} catch (Exception e) {
			e.printStackTrace();
			
		}finally {
			if (statement != null) {
				statement.close();
			}
			if (connection != null) {
				connection.close();
			}
		}
	}
