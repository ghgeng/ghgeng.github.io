JavaFX 运动模糊效果

提示

JavaFX MotionBlur
MotionBlur类似于高斯模糊效果。运动模糊效果也用于模糊节点。高斯模糊效果和MotionBlur效果之间的唯一区别是MotionBlur效果使用指定的角度来模糊节点。 顾名思义，通过应用这种效果，可以看到节点处于运动中。类javafx.scene.effect.MotionBlur代表了motionblur效果。为了生成适当的效果，需要实例化这个类。

# 属性

方法 描述
setAngle(Double value) 它表示运动效果的角度。它是一个双类型属性。
setInput(Effect value) 它表示效果的输入。它是一个Effect对象类型属性。
setRadius(Double radius) 它表示模糊核的半径。它是一个双类型属性。

# 示例1

package code.javafx.JavaFXEffects;

import javafx.application.Application;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.effect.GaussianBlur;
import javafx.scene.effect.MotionBlur;
import javafx.scene.paint.Color;
import javafx.scene.text.Font;
import javafx.scene.text.FontPosture;
import javafx.scene.text.FontWeight;
import javafx.scene.text.Text;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2023/9/20 17:39
 */
public class JavaFXMotionBlur extends Application {

    @Override
    public void start(Stage primaryStage) throws Exception {
        primaryStage.setTitle("JavaFXMotionBlur");

        Text text = new Text("Welcome to javaTpoint");
        text.setX(30);
        text.setY(100);
        text.setFont(Font.font("calibri", FontWeight.BOLD, FontPosture.ITALIC, 40));
        MotionBlur motion = new MotionBlur();
        motion.setAngle(20);
        motion.setRadius(10);
        text.setEffect(motion);

        Group root = new Group();
        root.getChildren().addAll(text);
        Scene scene = new Scene(root, 500, 350, Color.WHITE);
        primaryStage.setScene(scene);
        primaryStage.show();
    }

    public static void main(String[] args) {
        launch(args);
    }
}
