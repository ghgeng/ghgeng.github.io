# 自定义控件

  • 1、创建一个MyControl 集成Jpanel ，是一个控件也是一个窗口
  • 2、重写paintComponent() 方法
  • 3、设置颜色 g.setColor(new Color(255, 0, 0));

package code.swing.learn;

import javax.swing.*;
import java.awt.*;

/**
 * @author :ghgeng
 * @date 2023/6/10 12:49
 */
public class MyControl extends JPanel {

    @Override
    protected void paintComponent(Graphics g) {
        super.paintComponents(g);

        int width = this.getWidth();
        int height = this.getHeight();

        // 设置颜色, 和矩阵一起用
        g.setColor(new Color(255, 0, 0));
        // 矩阵
        g.fillRect(0, 0, width, height);

        // 设置透明度，把高度设置成0
        // Color 第四个参数，透明度从 255 -> 0
        g.setColor(new Color(0, 255, 255 , 255));
        g.fillRect(50,50,  50,50);

    }
}

package code.swing.learn;

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.text.SimpleDateFormat;
import java.util.Date;

/**
 * @author :ghgeng
 * @date 2023/6/3 9:40
 */
public class MySwing extends JFrame {

    private static JPanel root = new JPanel();

    public static void main(String[] args) {
        MySwing mySwing = new MySwing();
        mySwing.initMyFrame();
        mySwing._08();
    }

    public void initMyFrame() {
        this.setTitle("第一个窗口");
        // 当关闭窗口时 ，默认关闭程序
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        // 设置窗口(组件)大小
        this.setSize(400, 600);
        // 居中展示
        this.setLocationRelativeTo(null);
        // 设置面板，所有的组件都可以 .add() 放上去
        // 就好比有一个画板，画画之前放一张纸上去，大小可以自己设置
        this.setContentPane(root);


    }

    public void _08() {
        MyControl myControl = new MyControl();
        root.add(myControl);
        
        myControl.setPreferredSize(new Dimension(this.getWidth(), this.getHeight()));

        this.setVisible(true);
    }


}

# 绘制几何形状

  • 1、以fill*打头都是填充
  • 2、以draw*开头都是描边

    @Override
    protected void paintComponent(Graphics g) {
        super.paintComponents(g);

        // 设置透明度，把高度设置成0
        // Color 第四个参数，透明度从 255 -> 0
        g.setColor(new Color(255, 255, 255, 255));
        g.fillRect(0,0,  this.getWidth(),this.getHeight());

        // 绘制矩形
        g.setColor(new Color(139, 139, 139));
        g.fillRect(20,0,  50 ,50);

        // 绘制椭圆
        g.setColor(new Color(142, 218, 241, 255));
        g.fillOval(100,50,  100 ,50);

        // 绘制圆形
        g.setColor(new Color(142, 218, 241, 255));
        g.fillOval(100,100,  100 ,100);

        // 绘制扇形
        g.setColor(new Color(174, 241, 202, 255));
        // 从20开始，逆时针旋转140度，也可以从 -40读开始
        g.fillArc(150,200,  100 ,100, -20 , 140);
        // 以上都是填充 以fill*打头
        // 描边
        g.setColor(new Color(206, 160, 255, 255));
        g.drawRect(150, 300, 50, 100);

        // 绘制线段
        g.setColor(new Color(206, 160, 255, 255));
        // 从 x1,y1 坐标开始 到 x2,y2 点绘制
        g.drawLine(150, 300, 100, 100);
    }
