JavaFX 高斯模糊效果

提示

JavaFX GaussianBlur
高斯模糊效果非常类似于BoxBlur效果。两者之间的唯一区别是高斯模糊效果使用高斯卷积核来模糊节点。但是，JavaFX提供了JavaFX .scene.effect.GaussianBlur类来在节点上实现GaussianBlur。这个类需要实例化，以便在节点上应用适当的效果。

# 属性

方法 描述
setInput(Effect value) 这个属性是效应类型。它代表了效果的输入。
setRadius(Double value) 这个属性是双重类型的。它表示模糊核的半径。

# 示例1

package code.javafx.JavaFXEffects;

import javafx.application.Application;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.effect.BoxBlur;
import javafx.scene.effect.GaussianBlur;
import javafx.scene.paint.Color;
import javafx.scene.text.Font;
import javafx.scene.text.FontPosture;
import javafx.scene.text.FontWeight;
import javafx.scene.text.Text;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2023/9/20 17:39
 */
public class JavaFXGaussianBlur extends Application {

    @Override
    public void start(Stage primaryStage) throws Exception {
        primaryStage.setTitle("JavaFXGaussianBlur");

        Text text = new Text();
        text.setText("Welcome to JavaTpoint");
        text.setX(100);
        text.setY(100);
        text.setFont(Font.font("Calibri", FontWeight.BLACK, FontPosture.ITALIC, 20));
        text.setFill(Color.RED);
        text.setStroke(Color.BLACK);
        text.setUnderline(true);
        GaussianBlur g = new GaussianBlur();
        g.setRadius(5);
        text.setEffect(g);

        Group root = new Group();
        root.getChildren().addAll(text);
        Scene scene = new Scene(root, 500, 350, Color.WHITE);
        primaryStage.setScene(scene);
        primaryStage.show();
    }

    public static void main(String[] args) {
        launch(args);
    }
}
