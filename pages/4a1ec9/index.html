JavaFX 节点模糊

提示

JavaFX BoxBlur
JavaFX允许我们通过使用JavaFX模糊效果来使节点模糊。一般来说，模糊会使图像不清晰。JavaFX提供了JavaFX .scene.effect. boxblur类，为了将模糊效果应用到节点上，需要将其实例化。Box过滤器用于JavaFX中的BoxBlur效果。

# 属性

方法 描述
setHeight(double value) 这是一个双类型属性。它表示模糊效果的高度。
setWidth(double value) 这是一个双类型属性。它表示模糊效果的宽度。
setInput(Effect value) 此属性为Effect类型。这表示效果的输入。
setIterations(int value) 它表示模糊效果的重复次数。这是一个整数类型。

# 示例1

package code.javafx.JavaFXEffects;

import javafx.application.Application;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.effect.BoxBlur;
import javafx.scene.effect.Glow;
import javafx.scene.image.Image;
import javafx.scene.image.ImageView;
import javafx.scene.paint.Color;
import javafx.scene.text.Font;
import javafx.scene.text.FontPosture;
import javafx.scene.text.FontWeight;
import javafx.scene.text.Text;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2023/9/20 17:39
 */
public class JavaFXBoxBlur extends Application {

    @Override
    public void start(Stage primaryStage) throws Exception {
        primaryStage.setTitle("JavaFXBoxBlur");

        Text text = new Text();
        text.setText("Hello JavaFXBoxBlur");
        text.setX(100);
        text.setY(100);
        text.setFont(Font.font("Calibri",FontWeight.BLACK,FontPosture.ITALIC,20));
        text.setFill(Color.RED);
        text.setStroke(Color.BLACK);
        text.setUnderline(true);
        BoxBlur b = new BoxBlur();
        b.setHeight(5);
        b.setWidth(2);
        b.setIterations(1);
        text.setEffect(b);

        Group root = new Group();
        root.getChildren().addAll(text);
        Scene scene = new Scene(root, 500,350, Color.WHITE);
        primaryStage.setScene(scene);
        primaryStage.show();
    }

    public static void main(String[] args) {
        launch(args);
    }
}
