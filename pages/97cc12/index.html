JavaFX Line

# 如何创建一条线

  • 实例化类javafx.scene.shape.Line。
  • 设置类对象所需的属性。
  • 向组中添加类对象

# 示例1

package code.javafx;

import javafx.application.Application;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.shape.Line;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2023/9/20 17:27
 */
public class JavaFXLine extends Application {


    @Override
    public void start(Stage primaryStage) throws Exception {
        // 创建line
        Line line = new Line();
        line.setStartX(20);
        line.setEndX(80);
        line.setStartY(30);
        line.setEndY(100);
        
        Group group = new Group();
        group.getChildren().add(line);

        Scene scene = new Scene(group,300 ,300);
        primaryStage.setScene(scene);
        primaryStage.setTitle("JavaFXLine");
        primaryStage.show();

    }
    public static void main(String[] args) {
        launch(args);
    }
}

# 示例2

package code.javafx;

import javafx.application.Application;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.paint.Color;
import javafx.scene.shape.Line;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2023/9/20 17:27
 */
public class JavaFXLine extends Application {


    @Override
    public void start(Stage primaryStage) throws Exception {
        // TODO Auto-generated method stub
        primaryStage.setTitle("Line Drawing Examples");
        Line line1 = new Line(10, 50, 150, 50); //Line(startX,startY,endX,endY)
        Line line2 = new Line(10, 100, 150, 100);
        Line line3 = new Line(10, 50, 10, 100);
        Line line4 = new Line(150, 50, 150, 100);
        Group root = new Group();
        root.getChildren().addAll(line1, line2, line3, line4);
        Scene scene = new Scene(root, 300, 200, Color.GRAY);
        primaryStage.setScene(scene);
        primaryStage.show();
    }

    public static void main(String[] args) {
        launch(args);
    }
}
