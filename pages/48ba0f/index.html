JavaFX Quad Curve

提示

四次曲线是二阶的普通曲线。它与三次曲线的不同之处在于它不像三次曲线那样有两个控制点。它只有一个控制点(X,Y)。
在JavaFX中，我们可以实例化JavaFX .scene.shape. quadcurve类来创建Quad曲线。这个类包含在下表中定义的各种属性。该类还包含setter方法，可以根据我们的要求设置属性以获得四轴曲线。

# 示例

四次曲线

package code.javafx;

import javafx.application.Application;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.effect.DropShadow;
import javafx.scene.paint.Color;
import javafx.scene.shape.CubicCurve;
import javafx.scene.shape.QuadCurve;
import javafx.stage.Stage;

/**
 * @author :ghgeng
 * @date 2023/9/20 17:39
 */
public class JavaFXQuadCurve extends Application {

    @Override
    public void start(Stage primaryStage) throws Exception {
        primaryStage.setTitle("JavaFXQuadCurve");
        Group group = new Group();
        QuadCurve c = new QuadCurve();
        c.setStartX(70);
        c.setStartY(30);
        c.setControlX(250);
        c.setControlY(50);
        c.setFill(Color.RED);
        c.setEndX(250);
        c.setEndY(300);
        // 投影
        c.setEffect(new DropShadow());

        group.getChildren().addAll(c);
        Scene scene = new Scene(group, 300, 400, Color.ANTIQUEWHITE);
        primaryStage.setScene(scene);
        primaryStage.show();
    }


    public static void main(String[] args) {
        launch(args);
    }
}
