JavaFX CheckBox

# JavaFX CheckBox

提示

复选框用于向用户提供一个以上的选择。它可用于提示用户选择多个选项或用户希望选择多个选项的场景。 它不同于radiobutton，因为我们可以在一个场景中选择多个复选框。 实例化javafx.scene.control.CheckBox类以实现CheckBox。 使用以下代码行创建一个空白复选框。

# 示例一

package code.javafx.JavaFXUI;

import javafx.application.Application;
import javafx.scene.Scene;
import javafx.scene.control.CheckBox;
import javafx.scene.control.Label;
import javafx.scene.layout.HBox;
import javafx.stage.Stage;

public class CheckBoxTest extends Application {

    public static void main(String[] args) {
        launch(args);
    }

    @Override
    public void start(Stage primaryStage) throws Exception {
        // TODO Auto-generated method stub
        Label l = new Label("What do you listen:     ");
        CheckBox c1 = new CheckBox("Radio one");
        CheckBox c2 = new CheckBox("Radio Mirchi");
        CheckBox c3 = new CheckBox("Red FM");
        CheckBox c4 = new CheckBox("FM GOLD");

        c1.setOnAction( e ->{
            System.out.println("Radio one是否被选中: " +  c1.isSelected());
        });

        HBox root = new HBox();
        root.getChildren().addAll(l, c1, c2, c3, c4);
        root.setSpacing(5);
        Scene scene = new Scene(root, 800, 200);
        primaryStage.setScene(scene);
        primaryStage.setTitle("CheckBox Example");
        primaryStage.show();
    }
}
