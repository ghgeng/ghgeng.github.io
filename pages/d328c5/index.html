# 设置字体属性

package code.swing.learn;

import javax.swing.*;
import java.awt.*;

/**
 * @author :ghgeng
 * @date 2023/6/6 22:52
 */
public class MyLayoutManager extends JFrame {
    private static JPanel root = new JPanel();

    public static void main(String[] args) {
        MyLayoutManager mySwing = new MyLayoutManager();
        mySwing.initMyFrame();
        mySwing._01_color();
    }


    /**
     * 设置字体属性
     */
    public void _01_color() {
        JLabel jLabel = new JLabel("测试");
        jLabel.setText("Hello World");
        root.add(jLabel);

        // 设置字体样式和大小等属性
        jLabel.setFont(new Font("微软雅黑", Font.BOLD, 18));
        // 设置字体颜色
        jLabel.setForeground(new Color(255, 0, 0));
        // 设置背景色
        // 默认背景是透明显示的，所以 setOpaque()设置为不透明才行
        jLabel.setOpaque(true);
        jLabel.setBackground(new Color(0, 0, 255));

        // 控件大小
        jLabel.setPreferredSize(new Dimension(200, 100));
        // 文字对齐方式
        jLabel.setHorizontalAlignment(SwingConstants.CENTER);
    }
    public void initMyFrame() {
        this.setTitle("第一个窗口");
        // 当关闭窗口时 ，默认关闭程序
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        // 设置窗口(组件)大小
        this.setSize(400, 600);
        // 居中展示
        this.setLocationRelativeTo(null);
        // 设置面板，所有的组件都可以 .add() 放上去
        // 就好比有一个画板，画画之前放一张纸上去，大小可以自己设置
        this.setContentPane(root);

        this.setVisible(true);
    }
}

    /**
     * FlowLayout 流式布局

    public void _02_flowLayout() {
        FlowLayout flowLayout = new FlowLayout();
        root.setLayout(flowLayout);

        JLabel jLabel1 = new ColorLable("1", Color.red);
        JLabel jLabel2 = new ColorLable("2", Color.PINK);

        jLabel3.setPreferredSize(new Dimension(100, 100));

    }

    class ColorLable extends JLabel {
        public ColorLable(String text, Color color) {

            this.setOpaque(true);
            this.setBackground(color);
            this.setPreferredSize(new Dimension(60, 30));
            this.setHorizontalAlignment(SwingConstants.CENTER);
        }

    }

# BorderLayout 边界布局

边界布局管理器。它把 Container 按方位分为 5 个区域（东、西、南、北、中），每个区域放置一个组件。

    /**
     * BorderLayout 边界布局
     * 边界布局管理器。它把 Container 按方位分为 5 个区域（东、西、南、北、中），每个区域放置一个组件。
     */
    public void _03_flowLayout() {
        root.setLayout(new BorderLayout());
        ColorLable c1 = new ColorLable("1", Color.pink);
        ColorLable c2 = new ColorLable("2", Color.green);
        ColorLable c3 = new ColorLable("3", Color.MAGENTA);
        ColorLable c4 = new ColorLable("4", Color.red);
        ColorLable c5 = new ColorLable("5", Color.YELLOW);

        root.add(c1 , BorderLayout.NORTH);
        root.add(c2 , BorderLayout.EAST);
        root.add(c3 , BorderLayout.WEST);
        root.add(c4 , BorderLayout.SOUTH);
        root.add(c5 , BorderLayout.CENTER);

        // 调整c1大小
        c1.setPreferredSize(new Dimension(0, 100));
        c3.setPreferredSize(new Dimension(100, 0));

        this.setVisible(true);
    }

# 自定义布局

需要实现LayoutManager2 ，就像一个管家一样管理容器
接口方法中 layoutContainer 重新布局
调整窗口重新计算组件的大小

    /**
     * 自定义布局
     */
    public void _05() {

        root.setLayout(new LayoutManager2() {

            @Override
            public void addLayoutComponent(String name, Component comp) {

            }

            @Override
            public void removeLayoutComponent(Component comp) {

            }

            @Override
            public Dimension preferredLayoutSize(Container parent) {
                return null;
            }

            @Override
            public Dimension minimumLayoutSize(Container parent) {
                return null;
            }

            @Override
            public void layoutContainer(Container parent) {
                // 整个布局的宽度
                int width = parent.getWidth();
                // 整个布局的高度
                int height = parent.getHeight();
                // 获取所有子容器
                Component[] components = parent.getComponents();
                int x = 0;
                int y = 0;
                for (int i = 0; i < components.length; i++) {
                    Component component = components[i];
                    component.setBounds(x, y, 100, 100);
                    // 第二个组件位置
                    x += 100;
                    if (x > width) {
                        // 如果组件总宽度超过面包宽度， 此时重新布局，让组件往下移动
                        x = 0 ;
                        y += 100;
                    }
                }
                System.out.println("layoutContainer...");
            }

            @Override
            public void addLayoutComponent(Component comp, Object constraints) {

            }

            @Override
            public Dimension maximumLayoutSize(Container target) {
                return null;
            }

            @Override
            public float getLayoutAlignmentX(Container target) {
                return 0;
            }

            @Override
            public float getLayoutAlignmentY(Container target) {
                return 0;
            }

            @Override
            public void invalidateLayout(Container target) {
            }
        });
        ColorLable c1 = new ColorLable("1", Color.pink);
        ColorLable c2 = new ColorLable("2", Color.green);
        ColorLable c3 = new ColorLable("3", Color.MAGENTA);
        ColorLable c4 = new ColorLable("4", Color.red);
        ColorLable c5 = new ColorLable("5", Color.YELLOW);

        root.add(c1, BorderLayout.NORTH);
        root.add(c2, BorderLayout.EAST);
        root.add(c3, BorderLayout.WEST);
        root.add(c4, BorderLayout.SOUTH);
        root.add(c5, BorderLayout.CENTER);

        this.setVisible(true);
    }
