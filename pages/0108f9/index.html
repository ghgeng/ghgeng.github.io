Java JMenuBar JMenu AND JMenuItem

# 类声明

  • JMenuBar
    • public class JMenuBar extends JComponent implements MenuElement, Accessible
  • JMenu
    • public class JMenu extends JMenuItem implements MenuElement, Accessible
  • JMenuItem
    • public class JMenuItem extends AbstractButton implements Accessible, MenuElement

# 示例代码1：

  • JMenuItem and JMenu Example
package code.swing.learn;

import javax.swing.*;

/**
 * @author :ghgeng
 * @date 2023/7/12 22:07
 */
public class MenuTest {

    public static void main(String[] args) {
        JFrame j = new JFrame("MenuTest");
        // 创建两个菜单
        JMenu user = new JMenu("用户");
        JMenu list = new JMenu("列表");
        JMenu other = new JMenu("其他");
        // 创建多个菜单子项
        JMenuItem jMenuItem1 = new JMenuItem("item1");
        JMenuItem jMenuItem2 = new JMenuItem("item2");
        JMenuItem jMenuItem3 = new JMenuItem("item3");
        JMenuItem jMenuItem4 = new JMenuItem("item4");
        JMenuItem jMenuItem5 = new JMenuItem("item5");
        JMenuItem jMenuItem6 = new JMenuItem("item6");

        // 给user菜单添加子菜单项
        user.add(jMenuItem1);
        user.add(jMenuItem2);
        user.add(jMenuItem3);
        // 给list菜单添加子菜单
        list.add(jMenuItem4);
        list.add(jMenuItem5);
        // 添加其他菜单
        other.add(jMenuItem6);
        // 给user菜单添加子菜单
        user.add(list);

        JMenuBar jMenuBar = new JMenuBar();
        jMenuBar.add(user);
        jMenuBar.add(other);
        // 给容器添加菜单
        j.setJMenuBar(jMenuBar);

        // 显示窗口
        j.setSize(400, 400);
        j.setLayout(null);
        j.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        j.setLocationRelativeTo(null);
        j.setVisible(true);
    }
}

# 示例代码2：

  • 创建简单的记事本
package code.swing.learn;

import javax.swing.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;

/**
 * @author :ghgeng
 * @date 2023/7/12 22:07
 */
public class MenuTest implements ActionListener {

    private JTextArea jTextArea;
    private JMenuItem cutItem;
    private JMenuItem copyItem;
    private JMenuItem pasteItem;
    private JMenuItem selectAllItem;

    public MenuTest() {
        JFrame j = new JFrame("Notepad");
        // 创建两个菜单
        JMenu fileMenu = new JMenu("file");
        JMenu editMenu = new JMenu("edit");
        JMenu helpMenu = new JMenu("help");

        // 创建多个菜单子项
        JMenuItem flie1Item = new JMenuItem("flie1");
        fileMenu.add(flie1Item);
        // 创建edit子菜单
        cutItem = new JMenuItem("cut");
        copyItem = new JMenuItem("copy");
        pasteItem = new JMenuItem("paste");
        selectAllItem = new JMenuItem("selectAll");
        cutItem.addActionListener(this);
        copyItem.addActionListener(this);
        pasteItem.addActionListener(this);
        selectAllItem.addActionListener(this);
        // 添加edit子菜单
        editMenu.add(cutItem);
        editMenu.add(copyItem);
        editMenu.add(pasteItem);
        editMenu.add(selectAllItem);
        // 创建help子菜单
        JMenuItem helpItem = new JMenuItem("help");
        helpMenu.add(helpItem);

        JMenuBar jMenuBar = new JMenuBar();

        jMenuBar.add(fileMenu);
        jMenuBar.add(editMenu);
        jMenuBar.add(helpMenu);
        // 给容器添加菜单
        j.setJMenuBar(jMenuBar);

        // 创建文本域
        jTextArea = new JTextArea();
        jTextArea.setBounds(5, 5, 360, 320);
        // 文本域换行
        jTextArea.setLineWrap(true);
        j.add(jTextArea);
        // 显示窗口
        j.setSize(400, 400);
        j.setLayout(null);
        j.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        j.setLocationRelativeTo(null);
        j.setVisible(true);

    }

    public static void main(String[] args) {
        new MenuTest();
    }

    @Override
    public void actionPerformed(ActionEvent e) {
        // 剪切
        if (e.getSource() == cutItem) {
            jTextArea.cut();
        }
        // 复制
        if (e.getSource() == copyItem) {
            jTextArea.copy();
        }
        // 粘贴
        if (e.getSource() == pasteItem) {
            jTextArea.paste();
        }
        // 选中全部
        if (e.getSource() == selectAllItem) {
            jTextArea.selectAll();
        }
    }
}
