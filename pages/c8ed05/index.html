Java JOptionPane

# JOptionPane

JOptionPane类用于提供标准的对话框，如消息对话框、确认对话框和输入对话框。这些对话框用于显示信息或从用户获取输入。JOptionPane类继承了JComponent类。

# 常用构造器:

Constructor Description
JOptionPane() 它用于创建带有测试消息的JOptionPane。
JOptionPane(Object message) 它用于创建JOptionPane的实例来显示消息。
JOptionPane(Object message, int messageType 它用于创建JOptionPane的实例,以显示带有指定消息类型和默认选项的消息。

# 常见方法

Methods Description
JDialog createDialog(String title) 它被用来创建并返回一个新的无主对话与指定的标题。
static void showMessageDialog(Component parentComponent, Object message) 它被用来创建一个名为Message的信息对话框。
static void showMessageDialog(Component parentComponent, Object message, String title, int messageType) 它用于创建带有给定标题和消息类型的消息对话框。
static int showConfirmDialog(Component parentComponent, Object message) 它被用来创建一个带有选项的对话框 Yes, No and Cancel;有了标题,选择一个选项。
static String showInputDialog(Component parentComponent, Object message) 它用于显示一个问题消息对话框，请求父级为parentComponent的用户输入。
void setInputValue(Object newValue) 它用于设置被用户选择或输入的输入值。

package code.swing.learn;

import javax.swing.*;

/**
 * @author :ghgeng
 * @date 2023/7/11 23:05
 */
public class JOptionPaneText {

    public static void main(String[] args) {
        JFrame jFrame = new JFrame();

        jFrame.setLayout(null);
        jFrame.setSize(400, 400);
        jFrame.setLocationRelativeTo(null);
        jFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        jFrame.setVisible(true);
        JOptionPane.showMessageDialog(jFrame, "Message Text");
    }
}

# 示例代码3： showConfirmDialog()

package code.swing.learn;

import javax.swing.*;
import java.awt.event.WindowAdapter;
import java.awt.event.WindowEvent;
import java.awt.event.WindowListener;

/**
 * @author :ghgeng
 * @date 2023/7/11 23:05
 */
public class JOptionPaneText extends WindowAdapter {
    JFrame jFrame = new JFrame();

    JOptionPaneText() {
        jFrame.addWindowListener(this);
        jFrame.setLayout(null);
        jFrame.setSize(400, 400);
        jFrame.setLocationRelativeTo(null);
        jFrame.setDefaultCloseOperation(JFrame.DO_NOTHING_ON_CLOSE);
        jFrame.setVisible(true);
    }

    public static void main(String[] args) {
        new JOptionPaneText();
    }

    @Override
    public void windowClosing(WindowEvent e) {
        int i = JOptionPane.showConfirmDialog(jFrame, "Are you sure?");
        if (i == JOptionPane.YES_OPTION) {
            jFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        }
    }
}
